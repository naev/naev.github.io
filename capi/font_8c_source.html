218  tex = NULL;

220  for (i=0; i<array_size( stsh->tex ); i++) {
221  for (j=0; j<MAX_ROWS; j++) {
222  r = &stsh->tex->rows[j];
223  /* Not empty row and doesn't fit. */
224  if ((r->h != 0) && (r->h != ch->h))
225  continue;
226  if (r->h == ch->h) {
227  /* Fits in current row, so use that. */
228  if (r->x + ch->w <= stsh->tw) {
229  tex = &stsh->tex[i];

231  break;
232  }

235  }
236  if (r->h != 0)
237  continue;
238  /* First row. */
239  if (j==0) {
240  assert( ch->h <= stsh->th ); /* Would be ridiculously large character... */
241  r->h = ch->h;
242  tex = &stsh->tex[i];
243  gr = r;
244  break;
245  }
246  /* See if height fits to create a new row. */
247  lr = &stsh->tex->rows[j-1];
248  if (lr->y + lr->h + ch->h <= stsh->th) {
249  r->h = ch->h;
250  r->y = lr->y + lr->h;
251  tex = &stsh->tex[i];
252  gr = r;
253  }
254  break; /* Have to break here because either we added a new row or texture is full. */
255  }
256  if (gr != NULL)
257  break;
258  }
259

263  memset( tex, 0, sizeof(glFontTex) );
264

266  glGenTextures( 1, &tex->id );

270  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, stsh->magfilter);
271  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, stsh->minfilter);
272

279  GL_RED, GL_UNSIGNED_BYTE, NULL );

285  gr = &tex->rows[0];

307  stsh->vbo_vert_data = realloc( stsh->vbo_vert_data, 8*stsh->mvbo*sizeof(GLshort) );

319  * |
320  * |
321  * +----------------- glyph bottom

323  * v y

325  *
326  * +----+-------------> x

332  * \----/
333  * off_x
334  */

337  fh = (GLfloat) stsh->th;

339  ty = (GLfloat) gr->y / fh;

352  vbo_tex[ 5 ] = tyh;
353  vbo_tex[ 6 ] = txw; /* Bottom right. */
354  vbo_tex[ 7 ] = tyh;

356  vbo_vert[ 0 ] = vx; /* Top left. */
357  vbo_vert[ 1 ] = vy+vh;
358  vbo_vert[ 2 ] = vx+vw; /* Top right. */
359  vbo_vert[ 3 ] = vy+vh;
360  vbo_vert[ 4 ] = vx; /* Bottom left. */
361  vbo_vert[ 5 ] = vy;
362  vbo_vert[ 6 ] = vx+vw; /* Bottom right. */
363  vbo_vert[ 7 ] = vy;
364  /* Update vbos. */
365  gl_vboData( stsh->vbo_tex, sizeof(GLfloat)*n, stsh->vbo_tex_data );
366  gl_vboData( stsh->vbo_vert, sizeof(GLshort)*n, stsh->vbo_vert_data );

368  /* Add space for the new character. */
369  gr->x += ch->w;
370 
371  /* Save glyph data. */
372  glyph->vbo_id = (n-8)/2;
373  glyph->tex_index = tex - stsh->tex;

474 {

482  return 0;

485  gl_fontKernStart();
486  i = 0;
487  n = 0.;
488  while ((ch = u8_nextchar( text, &i ))) {
489  /* Ignore escape sequence. */

491  if (text[i] != '\0')
492  i++;

494  }

541  const int width, int *outw )
542 {

544  size_t i;
545  uint32_t ch;
546  GLfloat n;
547 
548  if (ft_font == NULL)
549  ft_font = &gl_defFont;
550  glFontStash *stsh = gl_fontGetStash( ft_font );
551 
552  /* limit size per line */
553  gl_fontKernStart();
554  lastspace = 0; /* last ' ' or '\n' in the text */
555  n = 0.; /* current width */
556  i = 0; /* current position */
557  while ((text[i] != '\n') && (text[i] != '\0')) {
558  /* Characters we should ignore. */
559  if (text[i] == '\t') {
560  i++;
561  continue;
562  }
563 
564  /* Ignore escape sequence. */
565  if (text[i] == FONT_COLOUR_CODE) {
566  if (text[i+1] != '\0')
567  i += 2;
568  else
569  i += 1;
570  continue;
571  }
572 
573  /* Save last space. */
574  if (text[i] == ' ') {
575  lastspace = i;
576  lastwidth = (int)round(n);
577  }
578 
579  ch = u8_nextchar( text, &i );
580  /* Unicode. */
581  glFontGlyph *glyph = gl_fontGetGlyph( stsh, ch );
582  adv_x = glyph==NULL ? 0 : gl_fontKernGlyph( stsh, ch, glyph ) + glyph->adv_x;
583  n += adv_x;
584 
585  /* Check if out of bounds. */
586  if (n > (GLfloat)width) {
587  if (lastspace > 0) {
588  if (outw != NULL)
589  *outw = lastwidth;
590  return lastspace;
591  }
592  /* Case we weren't able to write any whole words. */
593  if (outw != NULL)
594  *outw = (int)round(n - adv_x);
595  u8_dec( text, &i );
596  return i;
597  }
598  }
599 
600  /* Ran out of text so just return. */
601  if (outw != NULL)
602  *outw = (int)round(n);
603  return i;
604 }
