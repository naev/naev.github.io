145 void ovr_refresh (void)
146 {
147  double max_x, max_y;
148  int i, items, jumpitems;
149  Planet *pnt;
150  JumpPoint *jp;
151  const Vector2d **pos;
152  MapOverlayPos **mo;
153  char buf[STRMAX_SHORT];
154 
155  /* Must be open. */
156  if (!ovr_isOpen())
157  return;
158 
159  /* Calculate max size. */
160  items = 0;
161  pos = calloc(array_size(cur_system->jumps) + array_size(cur_system->planets), sizeof(Vector2d*));
162  mo = calloc(array_size(cur_system->jumps) + array_size(cur_system->planets), sizeof(MapOverlayPos*));
163  max_x = 0.;
164  max_y = 0.;
165  for (i=0; i<array_size(cur_system->jumps); i++) {
166  jp = &cur_system->jumps[i];
167  max_x = MAX( max_x, ABS(jp->pos.x) );
168  max_y = MAX( max_y, ABS(jp->pos.y) );
169  if (!jp_isUsable(jp) || !jp_isKnown(jp))
170  continue;
171  /* Initialize the map overlay stuff. */
172  snprintf( buf, sizeof(buf), "%s%s", jump_getSymbol(jp), sys_isKnown(jp->target) ? _(jp->target->name) : _("Unknown") );
173  pos[items] = &jp->pos;
174  mo[items] = &jp->mo;
175  mo[items]->radius = jumppoint_gfx->sw / 2.;
176  mo[items]->text_width = gl_printWidthRaw(&gl_smallFont, buf);
177  items++;
178  }
179  jumpitems = items;
180  for (i=0; i<array_size(cur_system->planets); i++) {
181  pnt = cur_system->planets[i];
182  max_x = MAX( max_x, ABS(pnt->pos.x) );
183  max_y = MAX( max_y, ABS(pnt->pos.y) );
184  if ((pnt->real != ASSET_REAL) || !planet_isKnown(pnt))
185  continue;
186  /* Initialize the map overlay stuff. */
187  snprintf( buf, sizeof(buf), "%s%s", planet_getSymbol(pnt), _(pnt->name) );
188  pos[items] = &pnt->pos;
189  mo[items] = &pnt->mo;
190  mo[items]->radius = pnt->radius / 2.; /* halved since it's awkwardly large if drawn to scale relative to the player. */
191  /* +2.0 represents a margin used by the SDF shader. */
192  mo[items]->text_width = gl_printWidthRaw( &gl_smallFont, buf );
193  items++;
194  }
195 
196  /* We need to calculate the radius of the rendering from the maximum radius of the system. */
197  ovr_res = 2. * 1.2 * MAX( max_x / map_overlay_width(), max_y / map_overlay_height() );
198  for (i=0; i<items; i++)
199  mo[i]->radius = MAX( 2.+mo[i]->radius / ovr_res, i<jumpitems ? 5. : 7.5 );
200 
201  /* Nothing in the system so we just set a default value. */
202  if (items == 0)
203  ovr_res = 50.;
204 
205  /* Compute text overlap and try to minimize it. */
206  ovr_optimizeLayout( items, pos, mo );
207 
208  /* Free the moos. */
209  free( mo );
210  free( pos );
211 }

217 static void ovr_optimizeLayout( int items, const Vector2d** pos, MapOverlayPos** mo )
218 {
219  int i, j, k, iter;
220  float cx, cy, r, sx, sy;
221  float x, y, w, h, mx, my, mw, mh;
222  float fx, fy, best, bx, by, val;
223  float *forces_xa, *forces_ya, *off_buffx, *off_buffy, *off_0x, *off_0y, old_bx, old_by, *off_dx, *off_dy;
224 
225  /* Parameters for the map overlay optimization. */
226  const int max_iters = 15;
227  const float kx = .015;
228  const float ky = .045;
229  const float eps_con = 1.3;
231  if (items <= 0)
232  return;
233 
234  /* Fix radii which fit together. */
235  MapOverlayRadiusConstraint cur, *fits = array_create(MapOverlayRadiusConstraint);
236  uint8_t *must_shrink = malloc( items );
237  for (cur.i=0; cur.i<items; cur.i++)
238  for (cur.j=cur.i+1; cur.j<items; cur.j++) {
239  cur.dist = hypot( pos[cur.i]->x - pos[cur.j]->x, pos[cur.i]->y - pos[cur.j]->y ) / ovr_res;
240  if (cur.dist < mo[cur.i]->radius + mo[cur.j]->radius)
241  array_push_back( &fits, cur );
242  }
243  while (array_size( fits ) > 0) {
244  float shrink_factor = 0;
245  memset( must_shrink, 0, items );
246  for (i = 0; i < array_size( fits ); i++) {
247  r = fits[i].dist / (mo[fits[i].i]->radius + mo[fits[i].j]->radius);
248  if (r >= 1)
249  array_erase( &fits, &fits[i], &fits[i+1] );
250  else {
251  shrink_factor = MAX( shrink_factor, r - FLT_EPSILON );
252  must_shrink[fits[i].i] = must_shrink[fits[i].j] = 1;
253  }
254  }
255  for (i=0; i<items; i++)
256  if (must_shrink[i])
257  mo[i]->radius *= shrink_factor;
258  }
259  free( must_shrink );
260  array_free( fits );
261 
262  /* Limit shrinkage. */
263  for (i=0; i<items; i++)
264  mo[i]->radius = MAX( mo[i]->radius, 4 );
265 
266  /* Initialization offset list. */
267  off_0x = calloc( items, sizeof(float) );
268  off_0y = calloc( items, sizeof(float) );
269 
270  /* Initialize all items. */
271  for (i=0; i<items; i++) {
272  /* Test to see what side is best to put the text on.
273  * We actually compute the text overlap also so hopefully it will alternate
274  * sides when stuff is clustered together. */
275 
276  x = pos[i]->x/ovr_res - ovr_text_pixbuf;
277  y = pos[i]->y/ovr_res - ovr_text_pixbuf;
278  w = mo[i]->text_width + 2*ovr_text_pixbuf;
279  h = gl_smallFont.h + 2*ovr_text_pixbuf;
280 
281  const float tx[4] = { mo[i]->radius+ovr_text_pixbuf+.1, -mo[i]->radius-.1-w, -mo[i]->text_width/2. , -mo[i]->text_width/2. };
282  const float ty[4] = { -gl_smallFont.h/2., -gl_smallFont.h/2., mo[i]->radius+ovr_text_pixbuf+.1, -mo[i]->radius-.1-h };
283 
284  /* Check all combinations. */
285  bx = 0.;
286  by = 0.;
287  best = HUGE_VALF;
288  for (k=0; k<4; k++) {
289  val = 0.;
290 
291  /* Test intersection with the planet indicators. */
292  for (j=0; j<items; j++) {
293  fx = fy = 0.;
294  mw = 2.*mo[j]->radius;
295  mh = mw;
296  mx = pos[j]->x/ovr_res - mw/2.;
297  my = pos[j]->y/ovr_res - mh/2.;
298 
299  force_collision( &fx, &fy, x+tx[k], y+ty[k], w, h, mx, my, mw, mh );
300 
301  val += ABS(fx) + ABS(fy);
302  }
303  /* Keep best. */
304  if (k == 0 || val < best) {
305  bx = tx[k];
306  by = ty[k];
307  best = val;
308  }
309  if (val==0.)
310  break;
311  }
312 
313  /* Store offsets. */
314  off_0x[i] = bx;
315  off_0y[i] = by;
316  }
317 
318  /* Uzawa optimization algorithm.
319  * We minimize the (weighted) L2 norm of vector of offsets and radius changes
320  * Under the constraint of no interpenetration
321  * As the algorithm is Uzawa, this constraint won't necessary be attained.
322  * This is similar to a contact problem is mechanics. */
323 
324  /* Initialize the matrix that stores the dual variables (forces applied between objects).
325  * matrix is column-major, this means it is interesting to store in each column the forces
326  * recieved by a given object. Then these forces are summed to obtain the total force on the object.
327  * Odd lines are forces from objects and Even lines from other texts. */
328 
329  forces_xa = calloc( 2*items*items, sizeof(float) );
330  forces_ya = calloc( 2*items*items, sizeof(float) );
331 
332  /* And buffer lists. */
333  off_buffx = calloc( items, sizeof(float) );
334  off_buffy = calloc( items, sizeof(float) );
335  off_dx = calloc( items, sizeof(float) );
336  off_dy = calloc( items, sizeof(float) );
337 
338  /* Main Uzawa Loop. */
339  for (iter=0; iter<max_iters; iter++) {
340  val = 0; /* This stores the stagnation indicator. */
341  for (i=0; i<items; i++) {
342  cx = pos[i]->x / ovr_res;
343  cy = pos[i]->y / ovr_res;
344  /* Compute the forces. */
345  ovr_refresh_uzawa_overlap(
346  forces_xa, forces_ya,
347  cx + off_dx[i] + off_0x[i] - ovr_text_pixbuf,
348  cy + off_dy[i] + off_0y[i] - ovr_text_pixbuf,
349  mo[i]->text_width + 2*ovr_text_pixbuf,
350  gl_smallFont.h + 2*ovr_text_pixbuf,
351  pos, mo, items, i, off_0x, off_0y, off_dx, off_dy );
352 
353  /* Do the sum. */
354  sx = sy = 0.;
355  for (j=0; j<2*items; j++) {
356  sx += forces_xa[2*items*i+j];
357  sy += forces_ya[2*items*i+j];
358  }
359 
360  /* Store old version of buffers. */
361  old_bx = off_buffx[i];
362  old_by = off_buffy[i];
363 
364  /* Update positions (in buffer). Diagonal stiffness. */
365  off_buffx[i] = kx * sx;
366  off_buffy[i] = ky * sy;
367 
368  val = MAX( val, ABS(old_bx-off_buffx[i]) + ABS(old_by-off_buffy[i]) );
369  }
370 
371  /* Offsets are actually updated once the first loop is over. */
372  for (i=0; i<items; i++) {
373  off_dx[i] = off_buffx[i];
374  off_dy[i] = off_buffy[i];
375  }
376 
377  /* Test stagnation. */
378  if (val <= eps_con)
379  break;
380  }
381 
382  /* Permanently add the initialization offset to total offset. */
383  for (i=0; i<items; i++) {
384  mo[i]->text_offx = off_dx[i] + off_0x[i];
385  mo[i]->text_offy = off_dy[i] + off_0y[i];
386  }
387 
388  /* Free the forces matrix and the various buffers. */
389  free( forces_xa );
390  free( forces_ya );
391  free( off_buffx );
392  free( off_buffy );
393  free( off_0x );
394  free( off_0y );
395  free( off_dx );
396  free( off_dy );
397 }

403 static void force_collision( float *ox, float *oy,
404  float x, float y, float w, float h,
405  float mx, float my, float mw, float mh )
406 {
407 
408  /* No contact because of y offset (+tolerance). */
409  if ((y+h < my+ovr_text_pixbuf) || (y+ovr_text_pixbuf > my+mh))
410  *ox = 0;
411  else {
412  /* Case A is left of B. */
413  if (x+.5*w < mx+.5*mw) {
414  *ox += mx-(x+w);
415  *ox = MIN(0., *ox);
416  }
417  /* Case A is to the right of B. */
418  else {
419  *ox += (mx+mw)-x;
420  *ox = MAX(0., *ox);
421  }
422  }
423 
424  /* No contact because of x offset (+tolerance). */
425  if ((x+w < mx+ovr_text_pixbuf) || (x+ovr_text_pixbuf > mx+mw))
426  *oy = 0;
427  else {
428  /* Case A is below B. */
429  if (y+.5*h < my+.5*mh) {
430  *oy += my-(y+h);
431  *oy = MIN(0., *oy);
432  }
433  /* Case A is above B. */
434  else {
435  *oy += (my+mh)-y;
436  *oy = MAX(0., *oy);
437  }
438  }
439 }

445 static void ovr_refresh_uzawa_overlap( float *forces_x, float *forces_y,
446  float x, float y, float w, float h, const Vector2d** pos,
447  MapOverlayPos** mo, int items, int self,
448  float *offx, float *offy, float *offdx, float *offdy )
449 {
450  int i;
451  float mx, my, mw, mh;
452  const float pb2 = ovr_text_pixbuf*2.;
453 
454  for (i=0; i<items; i++) {
455  /* Collisions with planet circles and jp triangles (odd indices). */
456  mw = 2.*mo[i]->radius;
457  mh = mw;
458  mx = pos[i]->x/ovr_res - mw/2.;
459  my = pos[i]->y/ovr_res - mh/2.;
460  force_collision( &forces_x[2*items*self+2*i+1], &forces_y[2*items*self+2*i+1], x, y, w, h, mx, my, mw, mh );
461 
462  if (i == self)
463  continue;
464 
465  /* Collisions with other texts (even indices) */
466  mw = mo[i]->text_width + pb2;
467  mh = gl_smallFont.h + pb2;
468  mx = pos[i]->x/ovr_res + offdx[i] + offx[i] - ovr_text_pixbuf;
469  my = pos[i]->y/ovr_res + offdy[i] + offy[i] - ovr_text_pixbuf;
470  force_collision( &forces_x[2*items*self+2*i], &forces_y[2*items*self+2*i], x, y, w, h, mx, my, mw, mh );
471  }
472 }

555 static int ovr_safelaneKnown( SafeLane *sf, Vector2d *posns[2] )
556 {
557  /* This is a bit asinine, but should be easily replaceable by decent code when we have a System Objects API.
558  * Specifically, a generic pos and isKnown test would clean this up nicely. */
559  Planet *pnt;
560  JumpPoint *jp;
561  int known = 1;
562  int j;
563  for (j=0; j<2; j++) {
564  switch(sf->point_type[j]) {
565  case SAFELANE_LOC_PLANET:
566  pnt = planet_getIndex( sf->point_id[j] );
567  posns[j] = &pnt->pos;
568  if (!planet_isKnown( pnt ))
569  known = 0;
570  break;
571  case SAFELANE_LOC_DEST_SYS:
572  jp = jump_getTarget( system_getIndex( sf->point_id[j] ), cur_system );
573  posns[j] = &jp->pos;
574  if (!jp_isKnown( jp ))
575  known = 0;
576  break;
577  default:
578  ERR( _("Invalid vertex type.") );
579  }
580  }
581  return known;
582 }

590 void ovr_render( double dt )
591 {
592  int i, j;
593  Pilot *const*pstk;
594  AsteroidAnchor *ast;
595  SafeLane *safelanes;
596  Planet *pnt;
597  JumpPoint *jp;
598  double w, h, res;
599  double x,y, r,detect;
600  double rx,ry, x2,y2, rw,rh;
601  glColour col;
602 
603  /* Must be open. */
604  if (!ovr_open)
605  return;
606 
607  /* Player must be alive. */
608  if (player_isFlag( PLAYER_DESTROYED ) || (player.p == NULL))
609  return;
610 
611  /* Default values. */
612  w = map_overlay_width();
613  h = map_overlay_height();
614  res = ovr_res;
615 
616  /* First render the background overlay. */
617  glColour c = { .r=0., .g=0., .b=0., .a= conf.map_overlay_opacity };
618  gl_renderRect( (double)gui_getMapOverlayBoundLeft(), (double)gui_getMapOverlayBoundRight(), w, h, &c );
619 
620  /* Render the safe lanes */
621  safelanes = safelanes_get( -1, 0, cur_system );
622  for (i=0; i<array_size(safelanes); i++) {
623  Vector2d *posns[2];
624  if (!ovr_safelaneKnown( &safelanes[i], posns ))
625  continue;
626 
627  /* Copy over alpha. */
628  if (ovr_render_safelanes != NULL)
629  safelanes[i].map_alpha = ovr_render_safelanes[i].map_alpha;
630 
631  if (safelanes[i].map_alpha < 1.0)
632  safelanes[i].map_alpha = MIN( safelanes[i].map_alpha+OVERLAY_FADEIN*dt, 1.0 );
633 
634  if (faction_isPlayerFriend( safelanes[i].faction ))
635  col = cFriend;
636  else if (faction_isPlayerEnemy( safelanes[i].faction ))
637  col = cHostile;
638  else
639  col = cNeutral;
640  col.a = 0.1 * safelanes[i].map_alpha;
641 
642  /* Get positions and stuff. */
643  map_overlayToScreenPos( &x, &y, posns[0]->x, posns[0]->y );
644  map_overlayToScreenPos( &x2, &y2, posns[1]->x, posns[1]->y );
645  rx = x2-x;
646  ry = y2-y;
647  r = atan2( ry, rx );
648  rw = MOD(rx,ry)/2.;
649  rh = 11.;
650 
651  /* Render. */
652  glUseProgram(shaders.safelanes.program);
653  gl_renderShader( x+rx/2., y+ry/2., rw, rh, r, &shaders.safelanes, &col, 1 );
654  }
655  array_free( ovr_render_safelanes );
