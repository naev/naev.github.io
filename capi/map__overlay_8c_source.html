106 int ovr_input( SDL_Event *event )
107 {
108  int mx, my;
109  double x, y;
110 
111  /* We only want mouse events. */
112  if (event->type != SDL_MOUSEBUTTONDOWN)
113  return 0;
114 
115  /* Player must not be NULL. */
116  if (player_isFlag(PLAYER_DESTROYED) || (player.p == NULL))
117  return 0;
118 
119  /* Player must not be dead. */
120  if (pilot_isFlag(player.p, PILOT_DEAD))
121  return 0;
122 
123  /* Mouse targeting only uses left and right buttons. */
124  if (event->button.button != SDL_BUTTON_LEFT &&
125  event->button.button != SDL_BUTTON_RIGHT)
126  return 0;
127 
128  /* Translate from window to screen. */
129  mx = event->button.x;
130  my = event->button.y;
131  gl_windowToScreenPos( &mx, &my, mx, my );
132 
133  /* Translate to space coords. */
134  x = ((double)mx - (double)map_overlay_center_x()) * ovr_res;
135  y = ((double)my - (double)map_overlay_center_y()) * ovr_res;
136 
137  return input_clickPos( event, x, y, 1., 10. * ovr_res, 15. * ovr_res );
138 }

146 void ovr_refresh (void)
147 {
148  double max_x, max_y;
149  int i, items, jumpitems;
150  Planet *pnt;
151  JumpPoint *jp;
152  const Vector2d **pos;
153  MapOverlayPos **mo;
154  MapOverlayPosOpt *moo;
155  char buf[STRMAX_SHORT];
156 
157  /* Must be open. */
158  if (!ovr_isOpen())
159  return;
160 
161  /* Calculate max size. */
162  items = 0;
163  pos = calloc(array_size(cur_system->jumps) + array_size(cur_system->planets), sizeof(Vector2d*));
164  mo = calloc(array_size(cur_system->jumps) + array_size(cur_system->planets), sizeof(MapOverlayPos*));
165  moo = calloc(array_size(cur_system->jumps) + array_size(cur_system->planets), sizeof(MapOverlayPosOpt));
166  max_x = 0.;
167  max_y = 0.;
168  for (i=0; i<array_size(cur_system->jumps); i++) {
169  jp = &cur_system->jumps[i];
170  max_x = MAX( max_x, ABS(jp->pos.x) );
171  max_y = MAX( max_y, ABS(jp->pos.y) );
172  if (!jp_isUsable(jp) || !jp_isKnown(jp))
173  continue;
174  /* Initialize the map overlay stuff. */
175  snprintf( buf, sizeof(buf), "%s%s", jump_getSymbol(jp), sys_isKnown(jp->target) ? _(jp->target->name) : _("Unknown") );
176  moo[items].text_width = gl_printWidthRaw(&gl_smallFont, buf);
177  pos[items] = &jp->pos;
178  mo[items] = &jp->mo;
179  mo[items]->radius = jumppoint_gfx->sw;
180  items++;
181  }
182  jumpitems = items;
183  for (i=0; i<array_size(cur_system->planets); i++) {
184  pnt = cur_system->planets[i];
185  max_x = MAX( max_x, ABS(pnt->pos.x) );
186  max_y = MAX( max_y, ABS(pnt->pos.y) );
187  if ((pnt->real != ASSET_REAL) || !planet_isKnown(pnt))
188  continue;
189  /* Initialize the map overlay stuff. */
190  snprintf( buf, sizeof(buf), "%s%s", planet_getSymbol(pnt), _(pnt->name) );
191  moo[items].text_width = gl_printWidthRaw( &gl_smallFont, buf );
192  pos[items] = &pnt->pos;
193  mo[items] = &pnt->mo;
194  mo[items]->radius = pnt->radius;
195  items++;
196  }
197 
198  /* We need to calculate the radius of the rendering from the maximum radius of the system. */
199  ovr_res = 2. * 1.2 * MAX( max_x / map_overlay_width(), max_y / map_overlay_height() );
200  for (i=0; i<items; i++)
201  mo[i]->radius = MAX( mo[i]->radius / ovr_res, i<jumpitems ? 10. : 15. );
202 
203  /* Compute text overlap and try to minimize it. */
204  ovr_optimizeLayout( items, pos, mo, moo, ovr_res );
205 
206  /* Free the moos. */
207  free( mo );
208  free( moo );
209  free( pos );
210 }

216 static void ovr_optimizeLayout( int items, const Vector2d** pos, MapOverlayPos** mo, MapOverlayPosOpt* moo, float res )
217 {
218  int i, iter, changed;
219  float cx,cy, ox,oy, r, off;
220 
221  /* Parameters for the map overlay optimization. */
222  const float update_rate = 0.015;
223  const int max_iters = 100;
224  const float pixbuf = 5.;
225  const float pixbuf_initial = 50;
226  const float position_threshold_x = 20.;
227  const float position_threshold_y = 10.;
228  const float position_weight = .1;
229  const float object_weight = 1.;
230  const float text_weight = 2.;
232  if (items <= 0)
233  return;
234 
235  /* Fix radii which fit together. */
236  MapOverlayRadiusConstraint cur, *fits = array_create(MapOverlayRadiusConstraint);
237  uint8_t *must_shrink = malloc( items );
238  for (cur.i=0; cur.i<items; cur.i++)
239  for (cur.j=cur.i+1; cur.j<items; cur.j++) {
240  cur.dist = hypot( pos[cur.i]->x - pos[cur.j]->x, pos[cur.i]->y - pos[cur.j]->y ) / res;
241  cur.dist *= 2; /* Oh, for the love of God, did someone make "radius" a diameter again? */
242  if (cur.dist < mo[cur.i]->radius + mo[cur.j]->radius)
243  array_push_back( &fits, cur );
244  }
245  while (array_size( fits ) > 0) {
246  float shrink_factor = 0;
247  memset( must_shrink, 0, items );
248  for (i = 0; i < array_size( fits ); i++)
249  {
250  r = fits[i].dist / (mo[fits[i].i]->radius + mo[fits[i].j]->radius);
251  if (r >= 1)
252  array_erase( &fits, &fits[i], &fits[i+1] );
253  else {
254  shrink_factor = MAX( shrink_factor, r - FLT_EPSILON );
255  must_shrink[fits[i].i] = must_shrink[fits[i].j] = 1;
256  }
257  }
258  for (i=0; i<items; i++)
259  if (must_shrink[i])
260  mo[i]->radius *= shrink_factor;
261  }
262  free( must_shrink );
263  array_free( fits );
264 
265  /* Initialize text positions to infinity. */
266  for (i=0; i<items; i++) {
267  mo[i]->text_offx = HUGE_VALF;
268  mo[i]->text_offy = HUGE_VALF;
269  }
270 
271  /* Initialize all items. */
272  for (i=0; i<items; i++) {
273  /* Test to see what side is best to put the text on.
274  * We actually compute the text overlap also so hopefully it will alternate
275  * sides when stuff is clustered together. */
276  cx = pos[i]->x / res;
277  cy = pos[i]->y / res;
278  ovr_init_position( &moo[i].text_offx_base, &moo[i].text_offy_base,
279  res, cx, cy, moo[i].text_width, gl_smallFont.h, pixbuf, pos, mo, moo, items, i,
280  pixbuf_initial, object_weight, text_weight );
281  moo[i].text_offx = moo[i].text_offx_base;
282  moo[i].text_offy = moo[i].text_offy_base;
283  /* Initialize mo. */
284  mo[i]->text_offx = moo[i].text_offx;
285  mo[i]->text_offy = moo[i].text_offy;
286  }
287 
288  /* Optimize over them. */
289  for (iter=0; iter<max_iters; iter++) {
290  changed = 0;
291  for (i=0; i<items; i++) {
292  cx = pos[i]->x / res;
293  cy = pos[i]->y / res;
294  /* Move text if overlap. */
295  if (ovr_refresh_compute_overlap( &ox, &oy, res, cx+mo[i]->text_offx, cy+mo[i]->text_offy, moo[i].text_width, gl_smallFont.h, pos, mo, moo, items, i, 0, pixbuf, object_weight, text_weight )) {
296  moo[i].text_offx += ox * update_rate;
297  moo[i].text_offy += 30 * oy * update_rate; /* Boost y offset as it's more likely to be the solution. */
298  changed = 1;
299  }
300 
301  /* Penalize offsets changes */
302  off = moo[i].text_offx_base - mo[i]->text_offx;
303  if (fabs(off) > position_threshold_x) {
304  off -= FSIGN(off) * position_threshold_x;
305  /* Regularization, my ass. This can kick the point straight through to the opposite side.
306  * That's not necessarily bad. If our base point forces a bad fit, may as well switch to another one.
307  * But we cannot just let the adjustment overshoot and grow without bound; it's hard to read a label
308  * located at (nan, nan). */
309  moo[i].text_offx += off * MIN( position_weight * fabs(off), 2. );
310  /* Embrace the possibility of switching sides (accidental simulated annealing?) and reset the base point. */
311  moo[i].text_offx_base *= FSIGN(moo[i].text_offx_base * moo[i].text_offx);
312  changed = 1;
313  }
314  off = moo[i].text_offy_base - mo[i]->text_offy;
315  if (fabs(off) > position_threshold_y) {
316  off -= FSIGN(off) * position_threshold_y;
317  moo[i].text_offy += off * MIN( position_weight * fabs(off), 2. );
318  moo[i].text_offy_base *= FSIGN(moo[i].text_offy_base * moo[i].text_offy);
319  changed = 1;
320  }
321 
322  /* Propagate updates. */
323  mo[i]->text_offx = moo[i].text_offx;
324  mo[i]->text_offy = moo[i].text_offy;
325  }
326  /* Converged (or unnecessary). */
327  if (!changed)
328  break;
329  }
330 }

336 static void ovr_init_position( float *px, float *py, float res, float x, float y, float w, float h,
337  float margin, const Vector2d** pos, MapOverlayPos** mo, MapOverlayPosOpt* moo, int items, int self,
338  float pixbuf, float object_weight, float text_weight )
339 {
340  int i;
341  float ox,oy, cx,cy, bx,by;
342  float off, val, best;
343 
344  off = mo[self]->radius/2.+margin*1.5;
345  /* Order is left -> right -> top -> bottom */
346  //float tx[8] = { off, -off-w, -w/2., -w/2., off, -off-w, off, -off-w };
347  //float ty[8] = { -h/2., -h/2., off, -off-h, off, off, -off-h, -off-h };
348  const float tx[4] = { off, -off-w, -w/2. , -w/2. };
349  const float ty[4] = { -h/2., -h/2., off, -off-h };
350 
351  /* Check all combinations. */
352  best = HUGE_VALF;
353  for (i=0; i<4; i++) {
354  cx = x + tx[i];
355  cy = y + ty[i];
356  ovr_refresh_compute_overlap( &ox, &oy, res, cx, cy, w, h, pos, mo, moo, items, self, 1, pixbuf, object_weight, text_weight );
357  val = pow2(ox)+pow2(oy);
358  /* Bias slightly toward the center, to avoid text going off the edge of the overlay. */
359  val -= 1 / (pow2(cx)+pow2(cy)+pow2(100));
360  /* Keep best. */
361  if (i == 0 || val < best) {
362  bx = tx[i];
363  by = ty[i];
364  best = val;
365  }
366  }
367 
368  *px = bx;
369  *py = by;
370 }

375 static int update_collision( float *ox, float *oy, float weight,
376  float x, float y, float w, float h,
377  float mx, float my, float mw, float mh )
378 {
379  /* No collision. */
380  if (((x+w) < mx) || (x > (mx+mw)))
381  return 0;
382  if (((y+h) < my) || (y > (my+mh)))
383  return 0;
384 
385  /* Case A is left of B. */
386  if (x < mx)
387  *ox += weight*(mx-(x+w));
388  /* Case A is to the right of B. */
389  else
390  *ox += weight*((mx+mw)-x);
391 
392  /* Case A is below B. */
393  if (y < my)
394  *oy += weight*(my-(y+h));
395  /* Case A is above B. */
396  else
397  *oy += weight*((my+mh)-y);
398 
399  return 1;
400 }

406 static int ovr_refresh_compute_overlap( float *ox, float *oy,
407  float res, float x, float y, float w, float h, const Vector2d** pos,
408  MapOverlayPos** mo, MapOverlayPosOpt* moo, int items, int self, int radius, float pixbuf,
409  float object_weight, float text_weight )
410 {
411  int i, collided;
412  float mx, my, mw, mh;
413  const float pb2 = pixbuf*2.;
414 
415  *ox = *oy = 0.;
416  collided = 0;
417 
418  for (i=0; i<items; i++) {
419  if (i != self || !radius) {
420  /* convert center coordinates to bottom left*/
421  mw = mo[i]->radius + pb2;
422  mh = mw;
423  mx = pos[i]->x/res - mw/2.;
424  my = pos[i]->y/res - mh/2.;
425  collided |= update_collision( ox, oy, object_weight, x, y, w, h, mx, my, mw, mh );
426  }
427  if (i != self || radius) {
428  /* no need to convert coordinates, just add pixbuf */
429  mw = moo[i].text_width + pb2;
430  mh = gl_smallFont.h + pb2;
431  mx = pos[i]->x/res + mo[i]->text_offx-pixbuf;
432  my = pos[i]->y/res + mo[i]->text_offy-pixbuf;
433  collided |= update_collision( ox, oy, text_weight, x, y, w, h, mx, my, mw, mh );
434  }
435  }
436 
437  return collided;
438 }

489 void ovr_render( double dt )
490 {
491  (void) dt;
492  int i, j;
493  Pilot *const*pstk;
494  AsteroidAnchor *ast;
495  double w, h, res;
496  double x,y;
497 
498  /* Must be open. */
499  if (!ovr_open)
500  return;
501 
502  /* Player must be alive. */
503  if (player_isFlag( PLAYER_DESTROYED ) || (player.p == NULL))
504  return;
505 
506  /* Default values. */
507  w = map_overlay_width();
508  h = map_overlay_height();
509  res = ovr_res;
510 
511  /* First render the background overlay. */
512  glColour c = { .r=0., .g=0., .b=0., .a= conf.map_overlay_opacity };
513  gl_renderRect( (double)gui_getMapOverlayBoundLeft(), (double)gui_getMapOverlayBoundRight(), w, h, &c );
514 
515  /* Render planets. */
516  for (i=0; i<array_size(cur_system->planets); i++)
517  if ((cur_system->planets[ i ]->real == ASSET_REAL) && (i != player.p->nav_planet))
518  gui_renderPlanet( i, RADAR_RECT, w, h, res, 1 );
519  if (player.p->nav_planet > -1)
520  gui_renderPlanet( player.p->nav_planet, RADAR_RECT, w, h, res, 1 );
521 
522  /* Render jump points. */
523  for (i=0; i<array_size(cur_system->jumps); i++)
524  if ((i != player.p->nav_hyperspace) && !jp_isFlag(&cur_system->jumps[i], JP_EXITONLY))
525  gui_renderJumpPoint( i, RADAR_RECT, w, h, res, 1 );
526  if (player.p->nav_hyperspace > -1)
527  gui_renderJumpPoint( player.p->nav_hyperspace, RADAR_RECT, w, h, res, 1 );
528 
529  /* Render pilots. */
530  pstk = pilot_getAll();
531  j = 0;
532  for (i=0; i<array_size(pstk); i++) {
533  if (pstk[i]->id == PLAYER_ID) /* Skip player. */
534  continue;
535  if (pstk[i]->id == player.p->target)
536  j = i;
537  else
538  gui_renderPilot( pstk[i], RADAR_RECT, w, h, res, 1 );
539  }
540  /* Render the targeted pilot */
541  if (j!=0)
542  gui_renderPilot( pstk[j], RADAR_RECT, w, h, res, 1 );
543 
544  /* Check if player has goto target. */
545  if (player_isFlag(PLAYER_AUTONAV) && (player.autonav == AUTONAV_POS_APPROACH)) {
546  x = player.autonav_pos.x / res + map_overlay_center_x();
547  y = player.autonav_pos.y / res + map_overlay_center_y();
548  gl_renderCross( x, y, 5., &cRadar_hilight );
549  gl_printMarkerRaw( &gl_smallFont, x+10., y-gl_smallFont.h/2., &cRadar_hilight, _("TARGET") );
550  }
551 
552  /* render the asteroids */
553  for (i=0; i<array_size(cur_system->asteroids); i++) {
554  ast = &cur_system->asteroids[i];
555  for (j=0; j<ast->nb; j++)
556  gui_renderAsteroid( &ast->asteroids[j], w, h, res, 1 );
557  }
558 
559  /* Render the player. */
560  gui_renderPlayer( res, 1 );
561 
562  /* Render markers. */
563  ovr_mrkRenderAll( res );
564 }
