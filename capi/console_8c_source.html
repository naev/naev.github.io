221 {
222  const char *fname;
223  char *buf;
224  size_t blen;
225  int n;
226 
227  /* Handle parameters. */
228  fname = luaL_checkstring(L, 1);
229  n = lua_gettop(L);
230 
231  /* Reset loaded buffer. */
232  if (cli_env != LUA_NOREF) {
233  nlua_getenv( cli_env, "_LOADED" );
234  if (lua_istable(L,-1)) {
235  lua_pushnil(L); /* t, nil */
236  while (lua_next(L, -2) != 0) { /* t, key, val */
237  lua_pop(L,1); /* t, key */
238  lua_pushvalue(L,-1); /* t, key, key */
239  lua_pushnil(L); /* t, key, key, nil */
240  lua_rawset(L,-4); /* t, key */
241  } /* t */
242  }
243  lua_pop(L,1); /* */
244  }
245 
246  /* Do the file from PHYSFS. */
247  buf = ndata_read( fname, &blen );
248  if (luaL_loadbuffer( L, buf, blen, fname ) != 0)
249  lua_error(L);
250  free( buf );
251 
252  /* Return the stuff. */
253  nlua_pushenv(cli_env);
254  lua_setfenv(L, -2);
255  lua_call(L, 0, LUA_MULTRET);

402  const char *match, *old;

404

406  if (old == NULL)

410  nlua_pushenv(cli_env);

412  for (i=0; str[i] != '\0'; i++) {
413  if (str[i] == '.' || str[i] == ':') {
414  str[i] = '\0';

416 
417  /* If not indexable, replace with blank table */
418  if (!lua_istable(naevL, -1)) {
419  if (luaL_getmetafield(naevL, -1, "__index")) {
420  if (lua_istable(naevL, -1)) {

428  lua_pop(naevL, 1);
429  lua_newtable(naevL);
430  }
431  }
432 
433  lua_remove(naevL, -2);
434  cur = str + i + 1;
435  /* Start over on other non-identifier character */
436  } else if (!isalnum(str[i]) && str[i] != '_') {
437  lua_pop(naevL, 1);

439  cur = str + i + 1;
440  }
441  }
442

446  if (lua_isstring(naevL, -2)) {
447  match = lua_tostring(naevL, -2);
448  if (strncmp(cur, match, strlen(cur)) == 0) {
449  new = malloc(strlen(old) + strlen(match) - strlen(cur) + 1);
450  strcpy(new, old);
451  strcat(new, match + strlen(cur));
452  window_setInput( wid, "inpInput", new);
453  free(new);
454  lua_pop(naevL, 2);
455  break;
456  }
457  }

459  }
460  }
461

546 static void cli_input( unsigned int wid, char *unused )
547 {
548  (void) unused;
549  int status, len;
550  char *str, *escaped;
551  char buf[CLI_MAX_INPUT+7];
552 
553  /* Get the input. */
554  str = window_getInput( wid, "inpInput" );
555 
556  /* Ignore useless stuff. */
557  if (str == NULL)
558  return;
559 
560  /* Put the message in the console. */
561  escaped = cli_escapeString( &len, str, strlen(str) );
562  snprintf( buf, CLI_MAX_INPUT+7, "#C%s %s#0",
563  cli_firstline ? "> " : ">>", escaped );
564  free(escaped);
565  cli_printCoreString( buf, 0 );
566 
567  /* Set up for concat. */
568  if (!cli_firstline) /* o */
569  lua_pushliteral(naevL, "\n"); /* o \n */
570 
571  /* Load the string. */
572  lua_pushstring( naevL, str ); /* s */
573 
574  /* Concat. */
575  if (!cli_firstline) /* o \n s */
576  lua_concat(naevL, 3); /* s */
577 
578  status = luaL_loadbuffer( naevL, lua_tostring(naevL,-1), lua_strlen(naevL,-1), "=cli" );
579 
580  /* String isn't proper Lua yet. */
581  if (status == LUA_ERRSYNTAX) {
582  size_t lmsg;
583  const char *msg = lua_tolstring(naevL, -1, &lmsg);
584  const char *tp = msg + lmsg - (sizeof(LUA_QL("<eof>")) - 1);
585  const char *s;
586  if (strstr(msg, LUA_QL("<eof>")) == tp) {
587  /* Pop the loaded buffer. */
588  lua_pop(naevL, 1);
589  cli_firstline = 0;
590  }
591  else {
592  /* Real error, spew message and break. */
593  s = lua_tostring(naevL, -1);
594  WARN( "%s", s );
595  cli_printCoreString( s, 1 );
596  lua_settop(naevL, 0);
597  cli_firstline = 1;
598  }
599  }
600 
601  /* Print results - all went well. */
602  else if (status == 0) {
603  lua_remove(naevL,1);
604 
605  nlua_pushenv(cli_env);
606  lua_setfenv(naevL, -2);
607 
608  if (nlua_pcall(cli_env, 0, LUA_MULTRET)) {
609  cli_printCoreString( lua_tostring(naevL, -1), 1 );
610  lua_pop(naevL, 1);
611  }
612 
613  if (lua_gettop(naevL) > 0) {
614  nlua_getenv(cli_env, "print");
615  lua_insert(naevL, 1);
616  if (lua_pcall(naevL, lua_gettop(naevL)-1, 0, 0) != 0)
617  cli_addMessage( _("Error printing results.") );
618  }
619 
620  /* Clear stack. */
621  lua_settop(naevL, 0);
622  cli_firstline = 1;
623  }
624 
625  /* Clear the box now. */
626  window_setInput( wid, "inpInput", NULL );
627 
628  /* Scroll to bottom */
629  cli_scroll_pos = -1;
630 }
