202 static int cli_script( lua_State *L )
203 {
204  const char *fname;
205  char *buf;
206  size_t blen;
207  int n;
208 
209  /* Handle parameters. */
210  fname = luaL_checkstring(L, 1);
211  n = lua_gettop(L);
212 
213  /* Reset loaded buffer. */
214  if (cli_env != LUA_NOREF) {
215  nlua_getenv( cli_env, "_LOADED" );
216  if (lua_istable(L,-1)) {
217  lua_pushnil(L); /* t, nil */
218  while (lua_next(L, -2) != 0) { /* t, key, val */
219  lua_pop(L,1); /* t, key */
220  lua_pushvalue(L,-1); /* t, key, key */
221  lua_pushnil(L); /* t, key, key, nil */
222  lua_rawset(L,-4); /* t, key */
223  } /* t */
224  }
225  lua_pop(L,1); /* */
226  }
227 
228  /* Do the file from PHYSFS. */
229  buf = ndata_read( fname, &blen );
230  if (luaL_loadbuffer( L, buf, blen, fname ) != 0)
231  lua_error(L);
232  free( buf );
233 
234  /* Return the stuff. */
235  nlua_pushenv(cli_env);
236  lua_setfenv(L, -2);
237  lua_call(L, 0, LUA_MULTRET);
238  return lua_gettop(L) - n;
239 }

381 
382  old = window_getInput( wid, "inpInput" );
383  str = strdup(old);

391

393  if (!lua_istable(naevL, -1)) {

403  lua_pop(naevL, 1);
404  lua_newtable(naevL);

406  }
407 
408  lua_remove(naevL, -2);
409  cur = str + i + 1;
410  /* Start over on other non-identifier character */
411  } else if (!isalnum(str[i]) && str[i] != '_') {
412  lua_pop(naevL, 1);

415  }
416  }
417

423  if (strncmp(cur, match, strlen(cur)) == 0) {
424  new = malloc(strlen(old) + strlen(match) - strlen(cur) + 1);
425  strcpy(new, old);
426  strcat(new, match + strlen(cur));
427  window_setInput( wid, "inpInput", new);
428  free(new);
429  lua_pop(naevL, 2);
430  break;
431  }
432  }
433  lua_pop(naevL, 1);
434  }
435  }

515 static void cli_input( unsigned int wid, const char *unused )
516 {
517  (void) unused;

519  const char *str;

522 
523  /* Get the input. */
524  str = window_getInput( wid, "inpInput" );
525 
526  /* Ignore useless stuff. */
527  if (str == NULL)
528  return;
529 
530  /* Put the message in the console. */
531  escaped = cli_escapeString( &len, str, strlen(str) );
532  snprintf( buf, CLI_MAX_INPUT+7, "#C%s %s#0",
533  cli_firstline ? "> " : ">>", escaped );
534  free(escaped);
535  cli_printCoreString( buf, 0 );
536 
537  /* Set up for concat. */
538  if (!cli_firstline) /* o */
539  lua_pushliteral(naevL, "\n"); /* o \n */
540 
541  /* Load the string. */
542  lua_pushstring( naevL, str ); /* s */
543 
544  /* Concat. */
545  if (!cli_firstline) /* o \n s */
546  lua_concat(naevL, 3); /* s */
547 
548  status = luaL_loadbuffer( naevL, lua_tostring(naevL,-1), lua_strlen(naevL,-1), "=cli" );
549 
550  /* String isn't proper Lua yet. */
551  if (status == LUA_ERRSYNTAX) {
552  size_t lmsg;
553  const char *msg = lua_tolstring(naevL, -1, &lmsg);
554  const char *tp = msg + lmsg - (sizeof(LUA_QL("<eof>")) - 1);
555  const char *s;
556  if (strstr(msg, LUA_QL("<eof>")) == tp) {
557  /* Pop the loaded buffer. */
558  lua_pop(naevL, 1);
559  cli_firstline = 0;
560  }
561  else {
562  /* Real error, spew message and break. */
563  s = lua_tostring(naevL, -1);
564  WARN( "%s", s );
565  cli_printCoreString( s, 1 );
566  lua_settop(naevL, 0);
567  cli_firstline = 1;
568  }
569  }
570 
571  /* Print results - all went well. */
572  else if (status == 0) {
573  lua_remove(naevL,1);
574 
575  nlua_pushenv(cli_env);
576  lua_setfenv(naevL, -2);
577 
578  if (nlua_pcall(cli_env, 0, LUA_MULTRET)) {
579  cli_printCoreString( lua_tostring(naevL, -1), 1 );
580  lua_pop(naevL, 1);
581  }
582

584  nlua_getenv(cli_env, "print");

587  cli_addMessage( _("Error printing results.") );

590  /* Clear stack. */
591  lua_settop(naevL, 0);
592  cli_firstline = 1;
593  }
594

596  window_setInput( wid, "inpInput", NULL );
597 
598  /* Scroll to bottom */
599  cli_scroll_pos = -1;
