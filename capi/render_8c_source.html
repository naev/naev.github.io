61 static void render_fbo( double dt, GLuint fbo, GLuint tex, PPShader *shader )
62 {
63  /* Have to consider alpha premultiply. */
64  glBlendFuncSeparate( GL_ONE, GL_ONE_MINUS_SRC_ALPHA, GL_ONE, GL_ONE_MINUS_SRC_ALPHA );
65 
66  glBindFramebuffer(GL_FRAMEBUFFER, fbo);
67 
68  glUseProgram( shader->program );
69 
70  /* Screen size. */
71  if (shader->love_ScreenSize >= 0)
72  /* TODO don't have to upload this every frame, only when resized... */
73  glUniform4f( shader->love_ScreenSize, SCREEN_W, SCREEN_H, 1., 0. );
74 
75  /* Time stuff. */
76  if (shader->u_time >= 0) {
77  shader->dt += dt;
78  glUniform1f( shader->u_time, shader->dt );
79  }
80 
81  /* Set up stuff .*/
82  glEnableVertexAttribArray( shader->VertexPosition );
83  gl_vboActivateAttribOffset( gl_squareVBO, shader->VertexPosition, 0, 2, GL_FLOAT, 0 );
84  if (shader->VertexTexCoord >= 0) {
85  glEnableVertexAttribArray( shader->VertexTexCoord );
86  gl_vboActivateAttribOffset( gl_squareVBO, shader->VertexTexCoord, 0, 2, GL_FLOAT, 0 );
87  }
88 
89  /* Set the texture(s). */
90  glBindTexture( GL_TEXTURE_2D, tex );
91  glUniform1i( shader->MainTex, 0 );
92  for (int i=0; i<array_size(shader->tex); i++) {
93  LuaTexture_t *t = &shader->tex[i];
94  glActiveTexture( t->active );
95  glBindTexture( GL_TEXTURE_2D, t->texid );
96  glUniform1i( t->uniform, t->value );
97  }
98  glActiveTexture( GL_TEXTURE0 );
99 
100  /* Set shader uniforms. */
101  gl_Matrix4_Uniform(shader->ClipSpaceFromLocal, gl_Matrix4_Ortho(0, 1, 1, 0, 1, -1));
102 
103  /* Draw. */
104  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4 );
105 
106  /* Clear state. */
107  glDisableVertexAttribArray( shader->VertexPosition );
108  if (shader->VertexTexCoord >= 0)
109  glDisableVertexAttribArray( shader->VertexTexCoord );
110  glUseProgram( 0 );
111 
112  /* Restore the default mode. */
113  glBlendFunc( GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA );
114 }

120 static void render_fbo_list( double dt, PPShader *list, int *current, int done )
121 {
122  PPShader *pp;
123  int i, cur, next;
124  cur = *current;
125 
126  /* Render all except the last post-process shader. */
127  for (i=0; i<array_size(list)-1; i++) {
128  pp = &list[i];
129  next = 1-cur;
130  /* Render cur to next. */
131  render_fbo( dt, gl_screen.fbo[next], gl_screen.fbo_tex[cur], pp );
132  cur = next;
133  }
134 
135  /* Final render is to the screen. */
136  pp = &list[i];
137  if (done) {
138  gl_screen.current_fbo = 0;
139  /* Do the render. */
140  render_fbo( dt, gl_screen.current_fbo, gl_screen.fbo_tex[cur], pp );
141  glBindFramebuffer(GL_FRAMEBUFFER, gl_screen.current_fbo);
142  return;
143 
144  }
145 
146  /* Draw the last shader. */
147  next = 1-cur;
148  render_fbo( dt, gl_screen.fbo[next], gl_screen.fbo_tex[cur], pp );
149  cur = next;
150 
151  /* Set the framebuffer again. */
152  gl_screen.current_fbo = gl_screen.fbo[cur];
153  glBindFramebuffer(GL_FRAMEBUFFER, gl_screen.current_fbo);
154 
155  /* Set the new current framebuffer. */
156  *current = cur;
157 }

179 {

190  if (pp_game || pp_gui || pp_final) {

192  glBindFramebuffer(GL_FRAMEBUFFER, 0);
193  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);

197  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
198 
199  /* Set to front buffer. */
200  gl_screen.current_fbo = gl_screen.fbo[cur];
201  }
202  else
203  gl_screen.current_fbo = 0;
204 
205  /* Bind and clear new drawing area. */
206  glBindFramebuffer(GL_FRAMEBUFFER, gl_screen.current_fbo);
207  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
208 
209  dt = (paused) ? 0. : game_dt;
210 
211  /* Background stuff */
212  space_render( real_dt ); /* Nebula looks really weird otherwise. */
213  hooks_run( "renderbg" );
214  planets_render();
215  spfx_render(SPFX_LAYER_BACK);
216  weapons_render(WEAPON_LAYER_BG, dt);
217  /* Middle stuff */
218  player_renderUnderlay(dt);
219  pilots_render(dt);
220  weapons_render(WEAPON_LAYER_FG, dt);
221  spfx_render(SPFX_LAYER_MIDDLE);
222  /* Foreground stuff */
223  player_render(dt);
224  spfx_render(SPFX_LAYER_FRONT);
225  space_renderOverlay(dt);
226  gui_renderReticles(dt);
227  pilots_renderOverlay(dt);
228  hooks_run( "renderfg" );
229 
230  /* Process game stuff only. */
231  if (pp_game)
232  render_fbo_list( dt, pp_shaders_list[PP_LAYER_GAME], &cur, !(pp_final || pp_gui) );
233 
234  /* GUi stuff. */
235  gui_render(dt);
236 
237  if (pp_gui)
238  render_fbo_list( dt, pp_shaders_list[PP_LAYER_GUI], &cur, !pp_final );
239 
240  /* Top stuff. */
241  ovr_render( real_dt ); /* Using real_dt is sort of a hack for now. */
242  display_fps( real_dt ); /* Exception using real_dt. */
243  toolkit_render();
244 
245  /* Final post-processing. */
246  if (pp_final)
247  render_fbo_list( dt, pp_shaders_list[PP_LAYER_FINAL], &cur, 1 );
248 
249  /* check error every loop */
250  gl_checkErr();
251 }

277 unsigned int render_postprocessAdd( LuaShader_t *shader, int layer, int priority )
278 {
279  PPShader *pp, **pp_shaders;
280  unsigned int id;
281 
282  /* Select the layer. */
283  if (layer < 0 || layer >= PP_LAYER_MAX) {
284  WARN(_("Unknown post-processing shader layer '%d'!"), layer);
285  return 0;
286  }
287  pp_shaders = &pp_shaders_list[layer];
288 
289  if (*pp_shaders==NULL)
290  *pp_shaders = array_create( PPShader );
291  pp = &array_grow( pp_shaders );
292  id = ++pp_shaders_id;
293  pp->id = id;
294  pp->priority = priority;
295  pp->program = shader->program;
296  pp->ClipSpaceFromLocal = shader->ClipSpaceFromLocal;
297  pp->MainTex = shader->MainTex;
298  pp->VertexPosition = shader->VertexPosition;
299  pp->VertexTexCoord = shader->VertexTexCoord;
300  if (shader->tex != NULL)
301  pp->tex = array_copy( LuaTexture_t, shader->tex );
302  else
303  pp->tex = NULL;
304  /* Special uniforms. */
305  pp->u_time = glGetUniformLocation( pp->program, "u_time" );
306  pp->love_ScreenSize = glGetUniformLocation( pp->program, "love_ScreenSize" );
307  pp->dt = 0.;
308 
309  /* Resort n case stuff is weird. */
310  qsort( *pp_shaders, array_size(*pp_shaders), sizeof(PPShader), ppshader_compare );
311 
312  gl_checkErr();
313 
314  return id;
315 }

324 int render_postprocessRm( unsigned int id )
325 {
326  int i, j, found;
327  PPShader *pp, *pp_shaders;
328 
329  found = -1;
330  for (j=0; j<PP_LAYER_MAX; j++) {
331  pp_shaders = pp_shaders_list[j];
332  for (i=0; i<array_size(pp_shaders); i++) {
333  pp = &pp_shaders[i];
334  if (pp->id != id)
335  continue;
336  found = i;
337  break;
338  }
339  if (found>=0)
340  break;
341  }
342  if (found==-1) {
343  WARN(_("Trying to remove non-existant post-processing shader with id '%d'!"), id);
344  return -1;
345  }
346 
347  /* No need to resort. */
348  array_erase( &pp_shaders_list[j], &pp_shaders_list[j][found], &pp_shaders_list[j][found+1] );
349  return 0;
350 }
