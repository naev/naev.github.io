75 void credits2str( char *str, credits_t credits, int decimals )
76 {
77  if (decimals < 0)
78  snprintf( str, ECON_CRED_STRLEN, _("%.*f ¤"), 0, (double)credits );
79  else if (credits >= 1000000000000000000LL)
80  snprintf( str, ECON_CRED_STRLEN, _("%.*f E¤"), decimals, (double)credits / 1000000000000000000. );
81  else if (credits >= 1000000000000000LL)
82  snprintf( str, ECON_CRED_STRLEN, _("%.*f P¤"), decimals, (double)credits / 1000000000000000. );
83  else if (credits >= 1000000000000LL)
84  snprintf( str, ECON_CRED_STRLEN, _("%.*f T¤"), decimals, (double)credits / 1000000000000. );
85  else if (credits >= 1000000000L)
86  snprintf( str, ECON_CRED_STRLEN, _("%.*f G¤"), decimals, (double)credits / 1000000000. );
87  else if (credits >= 1000000)
88  snprintf( str, ECON_CRED_STRLEN, _("%.*f M¤"), decimals, (double)credits / 1000000. );
89  else if (credits >= 1000)
90  snprintf( str, ECON_CRED_STRLEN, _("%.*f k¤"), decimals, (double)credits / 1000. );
91  else
92  snprintf (str, ECON_CRED_STRLEN, _("%.*f ¤"), decimals, (double)credits );
93 }

278 static int commodity_parse( Commodity *temp, xmlNodePtr parent )
279 {
280  xmlNodePtr node;
281  CommodityModifier *newdict;
282  /* Clear memory. */
283  memset( temp, 0, sizeof(Commodity) );
284  temp->period = 200;
285  temp->population_modifier = 0.;
286  temp->standard = 0;
287 
288  /* Parse body. */
289  node = parent->xmlChildrenNode;
290  do {
291  xml_onlyNodes(node);
292  xmlr_strd(node, "name", temp->name);
293  xmlr_strd(node, "description", temp->description);
294  xmlr_int(node, "price", temp->price);
295  if (xml_isNode(node,"gfx_space"))
296  temp->gfx_space = xml_parseTexture( node,
297  COMMODITY_GFX_PATH"space/%s", 1, 1, OPENGL_TEX_MIPMAPS );
298  if (xml_isNode(node,"gfx_store")) {
299  temp->gfx_store = xml_parseTexture( node,
300  COMMODITY_GFX_PATH"%s", 1, 1, OPENGL_TEX_MIPMAPS );
301  if (temp->gfx_store != NULL) {
302  } else {
303  temp->gfx_store = gl_newImage( COMMODITY_GFX_PATH"_default.webp", 0 );
304  }
305  continue;
306  }
307  if (xml_isNode(node, "standard")) {
308  temp->standard = 1;
309  continue;
310  }
311  xmlr_float(node, "population_modifier", temp->population_modifier);
312  xmlr_float(node, "period", temp->period);
313  if (xml_isNode(node, "planet_modifier")) {
314  newdict = malloc(sizeof(CommodityModifier));
315  newdict->next = temp->planet_modifier;
316  xmlr_attr_strd(node, "type", newdict->name);
317  newdict->value = xml_getFloat(node);
318  temp->planet_modifier = newdict;
319  continue;
320  }
321  if (xml_isNode(node, "faction_modifier")) {
322  newdict = malloc(sizeof(CommodityModifier));
323  newdict->next = temp->faction_modifier;
324  xmlr_attr_strd(node, "type", newdict->name);
325  newdict->value = xml_getFloat(node);
326  temp->faction_modifier = newdict;
327  }
328 
329  } while (xml_nextNode(node));
330  if (temp->name == NULL)
331  WARN( _("Commodity from %s has invalid or no name"), COMMODITY_DATA_PATH);
332  if ((temp->price > 0)) {
333  if (temp->gfx_store == NULL) {
334  WARN(_("No <gfx_store> node found, using default texture for commodity \"%s\""), temp->name);
335  temp->gfx_store = gl_newImage( COMMODITY_GFX_PATH"_default.webp", 0 );
336  }
337  if (temp->gfx_space == NULL)
338  temp->gfx_space = gl_newImage( COMMODITY_GFX_PATH"space/_default.webp", 0 );
339  }
340 
341 
342 
343 #if 0 /* shouldn't be needed atm */
344 #define MELEMENT(o,s) if (o) WARN( _("Commodity '%s' missing '"s"' element"), temp->name)
345  MELEMENT(temp->description==NULL,"description");
346  MELEMENT(temp->high==0,"high");
347  MELEMENT(temp->medium==0,"medium");
348  MELEMENT(temp->low==0,"low");
349 #undef MELEMENT

351 
352  return 0;
