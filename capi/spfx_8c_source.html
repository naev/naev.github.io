172 static int spfx_base_parse( SPFX_Base *temp, const char *filename )
173 {
174  xmlNodePtr node, cur, uniforms;
175  char *shadervert, *shaderfrag;
176  const char *name;
177  double x, y, z, w;
178  int ix, iy, iz, iw;
179  int isint;
180  GLint loc, dim;
181  xmlDocPtr doc;
182 
183  /* Load and read the data. */
184  doc = xml_parsePhysFS( filename );
185  if (doc == NULL)
186  return -1;
187 
188  /* Check to see if document exists. */
189  node = doc->xmlChildrenNode;
190  if (!xml_isNode(node,SPFX_XML_ID)) {
191  ERR( _("Malformed '%s' file: missing root element '%s'"), filename, SPFX_XML_ID);
192  return -1;
193  }
194 
195  /* Clear data. */
196  memset( temp, 0, sizeof(SPFX_Base) );
197  temp->shader = -1;
198  shadervert = NULL;
199  shaderfrag = NULL;
200  uniforms = NULL;
201 
202  xmlr_attr_strd( node, "name", temp->name );
203 
204  /* Extract the data. */
205  node = node->xmlChildrenNode;
206  do {
207  xml_onlyNodes(node);
208  xmlr_float(node, "anim", temp->anim);
209  xmlr_float(node, "ttl", temp->ttl);
210  if (xml_isNode(node,"gfx")) {
211  temp->gfx = xml_parseTexture( node,
212  SPFX_GFX_PATH"%s", 6, 5, 0 );
213  continue;
214  }
215 
216  if (xml_isNode(node,"shader")) {
217  cur = node->xmlChildrenNode;
218  do {
219  xml_onlyNodes(cur);
220  xmlr_strd(cur, "vert", shadervert);
221  xmlr_strd(cur, "frag", shaderfrag);
222  xmlr_float(cur, "size", temp->size);
223  if (xml_isNode(cur,"uniforms")) {
224  uniforms = cur;
225  continue;
226  }
227  } while (xml_nextNode(cur));
228  continue;
229  }
230  WARN(_("SPFX '%s' has unknown node '%s'."), temp->name, node->name);
231  } while (xml_nextNode(node));
232 
233  /* Convert from ms to s. */
234  temp->anim /= 1000.;
235  temp->ttl /= 1000.;
236  if (temp->ttl == 0.)
237  temp->ttl = temp->anim;
238 
239  /* Has shaders. */
240  if (shadervert != NULL && shaderfrag != NULL) {
241  temp->shader = gl_program_vert_frag( shadervert, shaderfrag );
242  temp->vertex = glGetAttribLocation( temp->shader, "vertex");
243  temp->projection = glGetUniformLocation( temp->shader, "projection");
244  temp->u_r = glGetUniformLocation( temp->shader, "u_r" );
245  temp->u_time = glGetUniformLocation( temp->shader, "u_time" );
246  temp->u_size = glGetUniformLocation( temp->shader, "u_size" );
247  if (uniforms != NULL) {
248  glUseProgram( temp->shader );
249  node = uniforms->xmlChildrenNode;
250  do {
251  xml_onlyNodes(node);
252  name = (char*)node->name;
253  loc = glGetUniformLocation( temp->shader, name );
254  if (loc < 0) {
255  WARN(_("SPFX '%s' is trying to set uniform '%s' not in shader!"), temp->name, name );
256  continue;
257  }
258  xmlr_attr_int_def(node,"int",isint,0);
259  /* Get dimension */
260  if (xmlHasProp(node,(xmlChar*)"w")) dim = 4;
261  else if (xmlHasProp(node,(xmlChar*)"z")) dim = 3;
262  else if (xmlHasProp(node,(xmlChar*)"y")) dim = 2;
263  else dim = 1;
264  /* Float values default to 0. */
265  if (isint) {
266  xmlr_attr_int(node, "x", ix );
267  xmlr_attr_int(node, "y", iy );
268  xmlr_attr_int(node, "z", iz );
269  xmlr_attr_int(node, "w", iw );
270  }
271  else {
272  xmlr_attr_float(node, "x", x );
273  xmlr_attr_float(node, "y", y );
274  xmlr_attr_float(node, "z", z );
275  xmlr_attr_float(node, "w", w );
276  }
277  switch (dim) {
278  case 1:
279  if (isint)
280  glUniform1i( loc, ix );
281  else
282  glUniform1f( loc, x );
283  break;
284  case 2:
285  if (isint)
286  glUniform2i( loc, ix, iy );
287  else
288  glUniform2f( loc, x, y );
289  break;
290  case 3:
291  if (isint)
292  glUniform3i( loc, ix, iy, iz );
293  else
294  glUniform3f( loc, x, y, z );
295  break;
296  case 4:
297  if (isint)
298  glUniform4i( loc, ix, iy, iz, iw );
299  else
300  glUniform4f( loc, x, y, z, w );
301  break;
302  default:
303  WARN(_("SPFX '%s' is trying to set uniform '%s' with '%d' dimensions!"), temp->name, name, dim );
304  continue;
305  }
306  } while (xml_nextNode(node));
307  glUseProgram( 0 );
308  }
309  gl_checkErr();
310  }
311 
312 #define MELEMENT(o,s) \
313  if (o) WARN( _("SPFX '%s' missing/invalid '%s' element"), temp->name, s)
314  MELEMENT(temp->anim==0.,"anim");
315  MELEMENT(temp->ttl==0.,"ttl");
316  MELEMENT(temp->gfx==NULL && (shadervert==NULL || shaderfrag==NULL),"gfx or shader");
317  MELEMENT(temp->shader>=0 && temp->size<=0., "shader/size");
318 #undef MELEMENT
319 
320  free(shadervert);
321  free(shaderfrag);
322 
323  /* Clean up. */
324  xmlFreeDoc(doc);
325 
326  return 0;
327 }

599 static void spfx_updateShake( double dt )
600 {
601  double mod, vmod, angle;
602  double force_x, force_y;
603  double dupdate;
604  int forced;
605 
606  /* Must still be on. */
607  if (shake_shader_pp_id == 0)
608  return;
609 
610  /* The shake decays over time */
611  forced = 0;
612  if (shake_force_mod > 0.) {
613  shake_force_mod -= SPFX_SHAKE_DECAY*dt;
614  if (shake_force_mod < 0.)
615  shake_force_mod = 0.;
616  else
617  forced = 1;
618  }
619  dupdate = dt*2.0;
620  shake_force_mean = dupdate*shake_force_mod + (1.0-dupdate)*shake_force_mean;
621 
622  /* See if it's settled down. */
623  mod = VMOD( shake_pos );
624  vmod = VMOD( shake_vel );
625  if (!forced && (mod < 0.01) && (vmod < 0.01)) {
626  render_postprocessRm( shake_shader_pp_id );
627  shake_shader_pp_id = 0;
628  if (fabs(shake_force_ang) > 1e3)
629  shake_force_ang = RNGF();
630  return;
631  }
632 
633  /* Calculate force. */
634  force_x = -SHAKE_K*shake_pos.x + -SHAKE_B*shake_vel.x;
635  force_y = -SHAKE_K*shake_pos.y + -SHAKE_B*shake_vel.y;
636 
637  /* Apply force if necessary. */
638  if (forced) {
639  shake_force_ang += dt;
640  angle = noise_simplex1( shake_noise, &shake_force_ang ) * 5.*M_PI;
641  force_x += shake_force_mod * cos(angle);
642  force_y += shake_force_mod * sin(angle);
643  }
644 
645  /* Update velocity. */
646  vect_cadd( &shake_vel, (1./SHAKE_MASS) * force_x * dt, (1./SHAKE_MASS) * force_y * dt );
647 
648  /* Update position. */
649  vect_cadd( &shake_pos, shake_vel.x * dt, shake_vel.y * dt );
650 
651  /* Set the uniform. */
652  glUseProgram( shaders.shake.program );
653  glUniform2f( shaders.shake.shake_pos, shake_pos.x / SCREEN_W, shake_pos.y / SCREEN_H );
654  glUniform2f( shaders.shake.shake_vel, shake_vel.x / SCREEN_W, shake_vel.y / SCREEN_H );
655  glUniform1f( shaders.shake.shake_force, shake_force_mean );
656  glUseProgram( 0 );
657 
658  gl_checkErr();
659 }

769 void spfx_trail_sample( Trail_spfx* trail, double x, double y, TrailMode mode, int force )
770 {
771  TrailPoint p;
772 
773  if (!force && trail->spec->style[mode].col.a <= 0.)
774  return;
775 
776  p.x = x;
777  p.y = y;
778  p.t = 1.;
779  p.mode = mode;
780 
781  /* The "back" of the trail should always reflect our most recent state. */
782  trail_back( trail ) = p;
783 
784  /* We may need to insert a control point, but not if our last sample was recent enough. */
785  if (!force && trail_size(trail) > 1 && trail_at( trail, trail->iwrite-2 ).t >= 1.-TRAIL_UPDATE_DT)
786  return;
787 
788  /* If the last time we inserted a control point was recent enough, we don't need a new one. */
789  if (trail_size(trail) == trail->capacity) {
790  /* Full! Double capacity, and make the elements contiguous. (We've made space to grow rightward.) */
791  trail->point_ringbuf = realloc( trail->point_ringbuf, 2 * trail->capacity * sizeof(TrailPoint) );

973 static void spfx_hapticRumble( double mod )
974 {
975  SDL_HapticEffect *efx;
976  double len, mag;
977 
978  /* Not active. */
979  if (haptic_rumble < 0)
980  return;
981 
982  /* Not time to update yet. */
983  if ((haptic_lastUpdate > 0.) || (shake_shader_pp_id==0) || (mod > SPFX_SHAKE_MAX/3.))
984  return;
985 
986  /* Stop the effect if it was playing. */
987  SDL_HapticStopEffect( haptic, haptic_rumble );
988

990  len = 1000. * shake_force_mod / SPFX_SHAKE_DECAY;
991  mag = 32767. * (shake_force_mod / SPFX_SHAKE_MAX);
992 
993  /* Update the effect. */
994  efx = &haptic_rumbleEffect;
995  efx->periodic.magnitude = (int16_t)mag;
996  efx->periodic.length = (uint32_t)len;

998  if (SDL_HapticUpdateEffect( haptic, haptic_rumble, &haptic_rumbleEffect ) < 0) {
999  WARN(_("Failed to update haptic effect: %s."), SDL_GetError());
1000  return;
1001  }
1002 
1003  /* Run the new effect. */
1004  SDL_HapticRunEffect( haptic, haptic_rumble, 1 );
1005 
1006  /* Set timer again. */
1007  haptic_lastUpdate += HAPTIC_UPDATE_INTERVAL;

1028 void spfx_render( const int layer )
1029 {

1031  int i;
1032  SPFX_Base *effect;
1033  int sx, sy;
1034  double time;
1035  Trail_spfx *trail;
1036 
1037  /* get the appropriate layer */
1038  switch (layer) {
1039  case SPFX_LAYER_FRONT:
1040  spfx_stack = spfx_stack_front;
1041  break;
1042 
1043  case SPFX_LAYER_MIDDLE:
1044  spfx_stack = spfx_stack_middle;
1045  break;
1046 
1047  case SPFX_LAYER_BACK:
1048  spfx_stack = spfx_stack_back;
1049  break;
1050 
1051  default:
1052  WARN(_("Rendering invalid SPFX layer."));
1053  return;
1054  }
1055 
1056  /* Trails are special (for now?). */
1057  if (layer == SPFX_LAYER_BACK)
1058  for (i=0; i<array_size(trail_spfx_stack); i++) {
1059  trail = trail_spfx_stack[i];
1060  if (!trail->ontop)
1061  spfx_trail_draw( trail );
1062  }
1063 
1064  /* Now render the layer */
1065  for (i=array_size(spfx_stack)-1; i>=0; i--) {
1066  spfx = &spfx_stack[i];
1067  effect = &spfx_effects[ spfx->effect ];
1068 
1069  /* Render shader. */
1070  if (effect->shader >= 0) {
1071  double x, y, z, s2;
1072  double w, h;
1073  gl_Matrix4 projection;
1074 
1075  /* Translate coords. */
1076  s2 = effect->size/2.;
1077  z = cam_getZoom();
1078  gl_gameToScreenCoords( &x, &y, spfx->pos.x-s2, spfx->pos.y-s2 );
1079  w = h = effect->size*z;
1080 
1081  /* Check if inbounds. */
1082  if ((x < -w) || (x > SCREEN_W+w) ||
1083  (y < -h) || (y > SCREEN_H+h))
1084  continue;
1085 
1086  /* Let's get to business. */
1087  glUseProgram( effect->shader );
1088 
1089  /* Set up the vertex. */
1090  projection = gl_view_matrix;
1091  projection = gl_Matrix4_Translate(projection, x, y, 0);
1092  projection = gl_Matrix4_Scale(projection, w, h, 1);
1093  glEnableVertexAttribArray( effect->vertex );
1094  gl_vboActivateAttribOffset( gl_squareVBO, effect->vertex,
1095  0, 2, GL_FLOAT, 0 );
1096 
1097  /* Set shader uniforms. */
1098  gl_Matrix4_Uniform(effect->projection, projection);
1099  glUniform1f(effect->u_time, spfx->time);
1100  glUniform1f(effect->u_r, spfx->unique);
1101  glUniform1f(effect->u_size, effect->size);
1102 
1103  /* Draw. */
1104  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4 );
1105 
1106  /* Clear state. */
1107  glDisableVertexAttribArray( shaders.texture.vertex );
1108 
1109  /* anything failed? */
1110  gl_checkErr();
1111 
1112  glUseProgram(0);
1113 
1114  }
1115  /* No shader. */
1116  else {
1117  /* Simplifies */
1118  sx = (int)effect->gfx->sx;
1119  sy = (int)effect->gfx->sy;
1120 
1121  if (!paused) { /* don't calculate frame if paused */
1122  time = 1. - fmod(spfx_stack[i].timer,effect->anim) / effect->anim;
1123  spfx_stack[i].lastframe = sx * sy * MIN(time, 1.);
1124  }
1125 
1126  /* Renders */
1127  gl_blitSprite( effect->gfx,
1128  VX(spfx_stack[i].pos), VY(spfx_stack[i].pos),
1129  spfx_stack[i].lastframe % sx,
1130  spfx_stack[i].lastframe / sx,
1131  NULL );
1132  }
1133  }

1192 static int trailSpec_load (void)
1193 {
1194  TrailSpec *tc, *parent;
1195  xmlNodePtr first, node;
1196  xmlDocPtr doc;
1197  char *name, *inherits;
1198 
1199  /* Load the data. */
1200  doc = xml_parsePhysFS( TRAIL_DATA_PATH );
1201  if (doc == NULL)
1202  return -1;
1203 
1204  /* Get the root node. */
1205  node = doc->xmlChildrenNode;
1206  if (!xml_isNode(node,"Trail_types")) {
1207  WARN( _("Malformed '%s' file: missing root element 'Trail_types'"), TRAIL_DATA_PATH);
1208  return -1;
1209  }
1210 
1211  /* Get the first node. */
1212  first = node->xmlChildrenNode; /* first event node */
1213  if (node == NULL) {
1214  WARN( _("Malformed '%s' file: does not contain elements"), TRAIL_DATA_PATH);
1215  return -1;
1216  }
1217 
1218  trail_spec_stack = array_create( TrailSpec );
1219 
1220  node = first;
1221  do {
1222  xml_onlyNodes( node );
1223  if (xml_isNode(node,"trail")) {
1224  tc = &array_grow( &trail_spec_stack );
1225  memset( tc, 0, sizeof(TrailSpec) );
1226  for(int i=0; i<MODE_MAX; i++)
1227  tc->style[i].thick = 1.;
1228  xmlr_attr_strd( node, "name", tc->name );
1229 
1230  /* Do the first pass for non-inheriting trails. */
1231  xmlr_attr_strd( node, "inherits", inherits );
1232  if (inherits == NULL)
1233  trailSpec_parse( node, tc );
1234  else
1235  free( inherits );
1236  }
1237  } while (xml_nextNode(node));
1238 
1239  /* Second pass to complete inheritance. */
1240  node = first;
1241  do {
1242  xml_onlyNodes( node );
1243  if (xml_isNode(node,"trail")) {
1244  /* Only interested in inherits. */
1245  xmlr_attr_strd( node, "inherits", inherits );
1246  if (inherits == NULL)
1247  continue;
1248  parent = trailSpec_getRaw( inherits );
1249 
1250  /* Get the style itself. */
1251  xmlr_attr_strd( node, "name", name );
1252  tc = trailSpec_getRaw( name );
1253 
1254  /* Make sure we found stuff. */
1255  if ((tc==NULL) || (parent==NULL)) {
1256  WARN(_("Trail '%s' that inherits from '%s' has missing reference!"), name, inherits );
1257  continue;
1258  }
1259  free( inherits );
1260  free( name );
1261 
1262  /* Set properties. */
1263  name = tc->name;
1264  memcpy( tc, parent, sizeof(TrailSpec) );
1265  tc->name = name;
1266 
1267  /* Load remaining properties (overrides parent). */
1268  trailSpec_parse( node, tc );
1269  }
1270  } while (xml_nextNode(node));
1271 
1272  for (tc=array_begin(trail_spec_stack); tc!=array_end(trail_spec_stack); tc++) {
1273  for(int i=0; i<MODE_MAX; i++)
1274  tc->style[i].thick *= tc->def_thick;
1275  }
1276  array_shrink(&trail_spec_stack);
1277 
1278  /* Clean up. */
1279  xmlFreeDoc(doc);
1280 
1281  return 0;
1282 }
