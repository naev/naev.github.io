498 {

500

503  base = delim==NULL ? strdup( buf ) : strndup( buf, delim-buf );
504 
505  ext = ".webp";
506  snprintf( str, sizeof(str), SHIP_GFX_PATH"%s/%s%s", base, buf, ext );

516  ship_loadEngineImage( temp, str, sx, sy );
517  if (temp->gfx_engine == NULL)

519  }

522  asprintf( &temp->gfx_comm, SHIP_GFX_PATH"%s/%s"SHIP_COMM"%s", base, buf, ext );
523  free( base );
524 
525  return 0;

678 {

684  int noengine;

687 
688  /* Clear memory. */
689  memset( temp, 0, sizeof(Ship) );
690 
691  /* Defaults. */
692  ss_statsInit( &temp->stats_array );
693 
694  /* Get name. */
695  xmlr_attr_strd( parent, "name", temp->name );
696  if (temp->name == NULL)
697  WARN( _("Ship in %s has invalid or no name"), SHIP_DATA_PATH );
698 
699  /* Datat that must be loaded first. */
700  node = parent->xmlChildrenNode;
701  do { /* load all the data */
702  xml_onlyNodes(node);
703  if (xml_isNode(node,"class")) {
704  xmlr_attr_strd( node, "display", temp->class_display );
705  temp->class = ship_classFromString( xml_get(node) );
706  continue;
707  }
708  } while (xml_nextNode(node));
709 
710  /* Default offsets for the engine. */
711  temp->trail_emitters = NULL;
712 
713  /* Load the rest of the data. */
714  node = parent->xmlChildrenNode;
715  do { /* load all the data */
716 
717  /* Only handle nodes. */
718  xml_onlyNodes(node);
719 
720  if (xml_isNode(node,"GFX")) {
721 
722  /* Get base graphic name. */
723  buf = xml_get(node);
724  if (buf==NULL) {
725  WARN(_("Ship '%s': GFX element is NULL"), temp->name);
726  continue;
727  }
728 
729  /* Get sprite size. */
730  xmlr_attr_int_def( node, "sx", sx, 8 );
731  xmlr_attr_int_def( node, "sy", sy, 8 );
732 
733  xmlr_attr_int(node, "noengine", noengine );
734 
735  /* Load the graphics. */
736  ship_loadGFX( temp, buf, sx, sy, !noengine );
737 
738  /* Load the polygon. */
739  ship_loadPLG( temp, buf, sx*sy );
740

742  if (array_size(temp->polygon) != sx*sy) {
743  WARN(_("Ship '%s': the number of collision polygons is wrong.\n \
744  npolygon = %i and sx*sy = %i"),
745  temp->name, array_size(temp->polygon), sx*sy);
746  }
747 
748  continue;
749  }
750 
751  if (xml_isNode(node,"gfx_space")) {
752 
753  /* Get path */
754  buf = xml_get(node);
755  if (buf==NULL) {
756  WARN(_("Ship '%s': gfx_space element is NULL"), temp->name);
757  continue;
758  }
759  snprintf( str, sizeof(str), GFX_PATH"%s", buf );
760 
761  /* Get sprite size. */
762  xmlr_attr_int_def( node, "sx", sx, 8 );
763  xmlr_attr_int_def( node, "sy", sy, 8 );
764 
765  /* Load the graphics. */
766  ship_loadSpaceImage( temp, str, sx, sy );
767 
768  continue;
769  }
770 
771  if (xml_isNode(node,"gfx_engine")) {
772 
773  /* Get path */
774  buf = xml_get(node);
775  if (buf==NULL) {
776  WARN(_("Ship '%s': gfx_engine element is NULL"), temp->name);
777  continue;
778  }
779  snprintf( str, sizeof(str), GFX_PATH"%s", buf );
780 
781  /* Get sprite size. */

783  xmlr_attr_int_def( node, "sy", sy, 8 );
784

786  ship_loadEngineImage( temp, str, sx, sy );
787 
788  continue;
789  }
790 
791  if (xml_isNode(node,"gfx_comm")) {
792  /* Get path */
793  buf = xml_get(node);
794  if (buf==NULL) {
795  WARN(_("Ship '%s': gfx_comm element is NULL"), temp->name);
796  continue;
797  }
798  snprintf( str, sizeof(str), GFX_PATH"%s", buf );
799  temp->gfx_comm = strdup(str);
800  continue;
801  }
802  if (xml_isNode(node,"gfx_overlays")) {
803  cur = node->children;
804  temp->gfx_overlays = array_create_size( glTexture*, 2 );
805  do {
806  xml_onlyNodes(cur);
807  if (xml_isNode(cur,"gfx_overlay"))
808  array_push_back( &temp->gfx_overlays,
809  xml_parseTexture( cur, OVERLAY_GFX_PATH"%s", 1, 1, OPENGL_TEX_MIPMAPS ) );
810  } while (xml_nextNode(cur));
811  continue;
812  }
813 
814  xmlr_strd(node,"GUI",temp->gui);
815  if (xml_isNode(node,"sound")) {
816  temp->sound = sound_get( xml_get(node) );
817  continue;
818  }
819  xmlr_strd(node,"base_type",temp->base_type);
820  if (xml_isNode(node,"class")) {
821  /* Already preemptively loaded, avoids warning. */
822  continue;
823  }
824  xmlr_float(node,"time_mod",temp->dt_default);
825  xmlr_long(node,"price",temp->price);
