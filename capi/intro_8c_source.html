72 static int intro_load( const char *text )
73 {
74  size_t intro_size;
75  char *intro_buf;
76  const char *cur_line;
77  char *rest_of_file;
78  char img_src[STRMAX_SHORT]; /* path to image to be displayed alongside text. */
79  glPrintLineIterator *iter;
80 
81  has_side_gfx = 0;
82 
83  /* Load text. */
84  intro_buf = ndata_read( text, &intro_size );
85 
86  /* Create intro font. */
87  gl_fontInit( &intro_font, _(FONT_MONOSPACE_PATH), conf.font_size_intro, FONT_PATH_PREFIX, 0 );
88 
89  /* Accumulate text into intro_lines. At each step, keep track of:
90  * * cur_line: text to go from the current input line, after word-wrapping.
91  * * rest_of_file: pointer to just after the current input line (or NULL if we're on the last one).
92  * We can start in a state with no word-wrap spill-over and "rest_of_file" pointing to the full input.
93  */
94  intro_lines = array_create( char* );
95  rest_of_file = intro_buf;
96  while (rest_of_file) {
97  cur_line = rest_of_file;
98  rest_of_file = strchr(cur_line, '\n');
99  /* If there's a next line, split the string and point rest_of_file to it. */
100  if (rest_of_file != NULL) {
101  /* Check for CRLF endings -- if present, zero both parts. */
102  if (rest_of_file > cur_line && *(rest_of_file-1) == '\r')
103  *(rest_of_file-1) = '\0';
104  *rest_of_file++ = '\0';
105  }
106 
107  if (cur_line[0] == '[' && sscanf( &cur_line[0], "[fadein %s", img_src ) == 1) {
108  /* an image to appear next to text. */
109  asprintf( &array_grow( &intro_lines ), "i%s", img_src );
110  /* Terminate the line where the closing bracket of [fadein ...] was. */
111  array_back( intro_lines )[ strlen( img_src ) ] = '\0';
112 
113  /* Mark that there are graphics. */
114  has_side_gfx = 1;
115  }
116  else if (cur_line[0] == '[' && strncmp( &cur_line[0], "[fadeout]", 9 ) == 0)
117  array_push_back( &intro_lines, strdup( "o" ) ); /* fade out the image next to the text. */
118  else if (cur_line[0] == '\0')
119  array_push_back( &intro_lines, strdup( "t" ) ); /* empty paragraph break. */
120  else { /* plain old text. */
121  iter = gl_printLineIteratorInit( &intro_font, _(cur_line), SCREEN_W - 2*SIDE_MARGIN - IMAGE_WIDTH );
122  while (gl_printLineIteratorNext( iter )) {
123  array_push_back( &intro_lines, malloc( iter->l_end - iter->l_begin + 2 ) );
124  array_back( intro_lines )[0] = 't';
125  strncpy( &array_back( intro_lines )[1], &iter->text[iter->l_begin], iter->l_end - iter->l_begin );
126  array_back( intro_lines )[iter->l_end - iter->l_begin + 1] = '\0';
127  }
128  gl_printLineIteratorFree( iter );
129  }
130  }
131 
132  /* Clean up. */
133  free(intro_buf);
134 
135  return 0;
136 }

197 static void intro_fade_image_in( intro_img_t *side, intro_img_t *transition,
198  const char *img_file )
199 {
200  if (NULL == side->tex)
201  load_image( side, img_file ); /* Simple fade-in. */
202  else {
203  /*
204  * Transition or on-deck. The difference is whether one image is
205  * replacing the other (transition), or whether the first must fade out
206  * completely before the second comes in (on-deck).
207  *
208  * We can determine which is the case by seeing whether [fadeout] has been
209  * called. I.e., is side->fade_rate < 0?
210  */
211  if (NULL != transition->tex) {
212  /* Scrolling is happening faster than fading... */
213  WARN( _("Intro scrolling too fast!") );
214  gl_freeTexture( transition->tex );
215  }
216  load_image( transition, img_file );
217  if (side->fade_rate < 0.0)
218  transition->fade_rate = 0.0; /* put an image on deck. */
219  else {
220  /* transition. */
221  side->fade_rate = -0.1; /* begin fading out. */
222  side->c.a = 0.99;
223  }
224  }
225 }

234 static void intro_event_handler( int *stop, double *offset, double *vel )
235 {
236  SDL_Event event; /* user key-press, mouse-push, etc. */
237 
238  while (SDL_PollEvent(&event)) {
239  if (event.type == SDL_WINDOWEVENT &&
240  event.window.event == SDL_WINDOWEVENT_RESIZED) {
241  naev_resize();
242  continue;
243  }
244 
245  if (event.type != SDL_KEYDOWN)
246  continue;
247 
248  /* Escape skips directly. */
249  if (event.key.keysym.sym == SDLK_ESCAPE) {
250  *stop = 1;
251  break;
252  }
253 
254  /* Slow down the text. */
255  else if (event.key.keysym.sym == SDLK_UP) {
256  *vel -= 8.0;
257  if (*vel < 0.0)
258  *vel = 0.0;
259  }
260 
261  /* Speed up the text. */
262  else if (event.key.keysym.sym == SDLK_DOWN) {
263  *vel += 8.0;
264  if (*vel > 100.0)
265  *vel = 100.0;
266  }
267 
268  /* Jump down. */
269  else if ((event.key.keysym.sym == SDLK_SPACE) ||
270  (event.key.keysym.sym == SDLK_RETURN))
271  *offset += 100;
272 
273  /* Jump up. */
274  else if (event.key.keysym.sym == SDLK_BACKSPACE)
275  *offset -= 100;
276 
277  /* User is clearly flailing on keyboard. */
278  else
279  *vel = 16.;
280  }
281 }

323 int intro_display( const char *text, const char *mus )
324 {
325  double offset; /* distance from bottom of the top line. */
326  double line_height; /* # pixels per line. */
327  int lines_per_screen; /* max appearing lines on the screen. */
328  char ** sb_arr; /* array of lines to render. */
329  int sb_index; /* Position in the line array. */
330  double vel = 16.; /* velocity: speed of text. */
331  int stop = 0; /* stop the intro. */
332  unsigned int tcur, tlast; /* timers. */
333  double delta; /* time diff from last render to this one. */
334  int line_index = 0; /* index into the big list of intro lines. */
335  intro_img_t side_image; /* image to go along with the text. */
336  intro_img_t transition; /* image for transitioning. */
337 
338  /* Load the introduction. */
339  if (intro_load(text) < 0)
340  return -1;
341 
342  /* Change music to intro music. */
343  if (mus != NULL)
344  music_choose(mus);
345 
346  /* We need to clear key repeat to avoid infinite loops. */
347  toolkit_clearKey();
348 
349  /* Do a few calculations to figure out how many lines can be present on the
350  screen at any given time. */
351  line_height = (double)intro_font.h * 1.3;
352  lines_per_screen = (int)(SCREEN_H / line_height + 1.5); /* round up + 1 */
353 
354  sb_arr = calloc( lines_per_screen, sizeof( char * ) );
355  sb_index = 0;
356 
357  /* Force the first line to be loaded immediately. */
358  offset = line_height;
359 
360  /* Unset the side image. */
361  initialize_image( &side_image );
362  initialize_image( &transition );
363 
364  tlast = SDL_GetTicks();
365  while (!stop) {
366  tcur = SDL_GetTicks();
367  delta = (double)(tcur - tlast) / 1000.;
368  tlast = tcur;
369 
370  /* Increment position. */
371  offset += vel * delta;
372  while (! (offset < line_height)) {
373  /* One line has scrolled off, and another one on. */
374 
375  if (line_index < array_size(intro_lines)) {
376  switch (intro_lines[line_index][0]) {
377  case 't': /* plain ol' text. */
378  sb_arr[ sb_index ] = &intro_lines[ line_index ][ 1 ];
379  offset -= line_height;
380  sb_index = ( sb_index + 1 ) % lines_per_screen;
381  break;
382  case 'i': /* fade in image. */
383  intro_fade_image_in( &side_image, &transition,
384  &intro_lines[line_index][1] );
385  break;
386  case 'o': /* fade out image. */
387  if (NULL == side_image.tex) {
388  WARN(_("Tried to fade out without an image.") );
389  break;
390  }
391  side_image.fade_rate = -0.1;
392  side_image.c.a = 0.99;
393  break;
394  default: /* unknown. */
395  break;
396  }
397  ++line_index;
398  } else {
399  sb_arr[ sb_index ] = NULL;
400  offset -= line_height;
401  sb_index = ( sb_index + 1 ) % lines_per_screen;
402  }
403  } /* while (offset > line_height) */
404 
405  /* Fade the side image. */
406  if (side_image.tex != NULL && side_image.c.a < 1.0) {
407  side_image.c.a += delta * vel * side_image.fade_rate;
408 
409  if (transition.tex != NULL && transition.fade_rate > 0.0)
410  transition.c.a += delta * vel * transition.fade_rate;
411 
412  if (side_image.c.a > 1.0) {
413  /* Faded in... */
414  side_image.c.a = 1.0;
415  side_image.fade_rate = 0.0;
416  }
417  else if (side_image.c.a < 0.0) {
418  /* Faded out... */
419  gl_freeTexture( side_image.tex );
420  if (transition.tex != NULL) {
421  side_image.tex = transition.tex;
422  side_image.c.a = transition.c.a;
423  side_image.w = transition.w;
424  side_image.h = transition.h;
425  side_image.y = transition.y;
426  side_image.fade_rate = 0.1;
427  transition.tex = NULL;
428  transition.c.a = 1.0;
429  }
430  else {
431  side_image.c.a = 1.0;
432  side_image.tex = NULL;
433  side_image.fade_rate = 0.0;
434  }
435  }
436  }
437 
438  /* Clear stuff. */
439  glClear(GL_COLOR_BUFFER_BIT);
440 
441  /* Only thing we actually care about updating is music. */
442  music_update( 0. );
443 
444  /* Draw text. */
445  stop = intro_draw_text( sb_arr, lines_per_screen, sb_index, offset, line_height );
446 
447  if (NULL != side_image.tex)
448  /* Draw the image next to the text. */
449  gl_renderScale( side_image.tex, side_image.x, side_image.y,
450  side_image.w, side_image.h, &side_image.c );
451 
452  if (NULL != transition.tex && transition.c.a > 0.0)
453  /* Draw the image in transition. */
454  gl_renderScale( transition.tex, transition.x, transition.y,
455  transition.w, transition.h, &transition.c );
456 
457  /* Display stuff. */
458  SDL_GL_SwapWindow( gl_screen.window );
459 
460 
461  SDL_Delay(10); /* No need to burn CPU. */
462 
463  /* Handle user events. */
464  intro_event_handler( &stop, &offset, &vel );
465 
466  } /* while (!stop) */
467 
468  /* free malloc'd memory. */
469  free( sb_arr );
470  gl_freeTexture( side_image.tex );
471  gl_freeTexture( transition.tex );
472 
473  /* Stop music, normal music will start shortly after. */
474  music_stop();
475 
476  /* Clean up after the introduction. */
477  intro_cleanup();
478 
479  return 0;
480 }
