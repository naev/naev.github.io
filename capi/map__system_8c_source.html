268 static void map_system_render( double bx, double by, double w, double h, void *data )
269 {
270  (void) data;
271  int i,j;
272  double iw, ih;
273  StarSystem *sys=cur_sys_sel;
274  Planet *p;
275  static int phase=0;
276  glColour ccol;
277  char buf[STRMAX_SHORT];
278  int cnt;
279  double ast_nb, ast_area;
280  double f;
281  int hasPresence = 0;
282  double unknownPresence = 0;
283  char t;
284  int txtHeight;
285  const glTexture *logo;
286  int offset;
287 
288  phase++;
289 
290  if (phase > 150) {
291  phase = 0;
292  starCnt++;
293  if ( starCnt >= (unsigned int) array_size( bgImages ) ) {
294  if ( array_size( bgImages ) <= 1)
295  starCnt = 0;
296  else
297  starCnt = 1;
298  }
299  }
300  /* background */
301  gl_renderRect( bx, by, w, h, &cBlack );
302 
303  j=0;
304  offset = h - pitch*nshow;
305  for ( i=0; i<array_size(sys->planets); i++ ) {
306  p=sys->planets[i];
307  j++;
308  if ( p->gfx_space == NULL) {
309  WARN( _("No gfx for %s...\n"),p->name );
310  } else {
311  ih=pitch;
312  iw = ih;
313  if ( p->gfx_space->w > p->gfx_space->h )
314  ih = ih * p->gfx_space->h / p->gfx_space->w;
315  else if ( p->gfx_space->w < p->gfx_space->h )
316  iw = iw * p->gfx_space->w / p->gfx_space->h;
317  gl_renderScale( p->gfx_space, bx+2, by+(nshow-j-1)*pitch + (pitch-ih)/2 + offset, iw, ih, &cWhite );
318  }
319  gl_printRaw( &gl_smallFont, bx + 5 + pitch, by + (nshow-j-0.5)*pitch + offset,
320  (cur_planet_sel == j ? &cFontGreen : &cFontWhite), -1., _(p->name) );
321  }
322  /* draw the star */
323  ih=pitch;
324  iw=ih;
325  if ( array_size( bgImages ) > 0 ) {
326  if ( bgImages[starCnt]->w > bgImages[starCnt]->h )
327  ih = ih * bgImages[starCnt]->h / bgImages[starCnt]->w;
328  else if ( bgImages[starCnt]->w < bgImages[starCnt]->h )
329  iw = iw * bgImages[starCnt]->w / bgImages[starCnt]->h;
330  ccol.r=ccol.g=ccol.b=ccol.a=1;
331  if ( phase > 120 && array_size( bgImages ) > 2 )
332  ccol.a = cos ( (phase-121)/30. *M_PI/2.);
333  gl_renderScale( bgImages[starCnt], bx+2 , by+(nshow-1)*pitch + (pitch-ih)/2 + offset, iw , ih, &ccol );
334  if ( phase > 120 && array_size( bgImages ) > 2) {
335  /* fade in the next star */
336  ih=pitch;
337  iw=ih;
338  i = starCnt + 1;
339  if ( i >= array_size( bgImages ) ) {
340  if ( array_size( bgImages ) <= 1 )
341  i=0;
342  else
343  i=1;
344  }
345  if ( bgImages[i]->w > bgImages[i]->h )
346  ih = ih * bgImages[i]->h / bgImages[i]->w;
347  else if ( bgImages[i]->w < bgImages[i]->h )
348  iw = iw * bgImages[i]->w / bgImages[i]->h;
349  ccol.a = 1 - ccol.a;
350  gl_renderScale( bgImages[i], bx+2, by+(nshow-1)*pitch + (pitch-ih)/2 + offset, iw, ih, &ccol );
351  }
352  } else {
353  /* no nebula or star images - probably due to nebula */
354  txtHeight=gl_printHeightRaw( &gl_smallFont,pitch,_("Obscured by the nebula") );
355  gl_printTextRaw( &gl_smallFont, pitch, txtHeight, (bx+2),
356  (by + (nshow-0.5)*pitch + offset), 0, &cFontRed, -1., _("Obscured by the nebula") );
357  }
358  gl_printRaw( &gl_smallFont, bx + 5 + pitch, by + (nshow-0.5)*pitch + offset,
359  (cur_planet_sel == 0 ? &cFontGreen : &cFontWhite), -1., _(sys->name) );
360  if ( cur_planet_sel == 0 && array_size( bgImages ) > 0 ) {
361  /* make use of space to draw a nice nebula */
362  double imgw,imgh;
363  iw = w - 50 - pitch - nameWidth;
364  ih = h - 110;
365  imgw = bgImages[0]->w;
366  imgh = bgImages[0]->h;
367  if ( (ih * imgw) / imgh > iw ) {
368  /* image is wider per height than the space allows - use all width */
369  int newih = (int)((iw * imgh) / imgw);
370  gl_renderScale( bgImages[0], bx + 10 + pitch + nameWidth, by + (ih-newih)/2, iw, newih, &cWhite );
371  } else {
372  /* image is higher, so use all height. */
373  int newiw = (int)((ih * imgw) / imgh);
374  gl_renderScale( bgImages[0], bx + 10 + pitch + nameWidth + (iw-newiw)/2, by, newiw, ih, &cWhite );
375  }
376  }
377  /* draw marker around currently selected planet */
378  ccol.r=0; ccol.g=0.6+0.4*sin( phase/150.*2*M_PI ); ccol.b=0; ccol.a=1;
379  ih=15;
380  iw=3;
381  gl_renderRect( bx+1, by+(nshow-cur_planet_sel-1)*pitch + offset, iw, ih, &ccol );
382  gl_renderRect( bx+1, by+(nshow-cur_planet_sel)*pitch-ih + offset, iw, ih, &ccol );
383  gl_renderRect( bx+pitch+3-iw, by+(nshow-cur_planet_sel-1)*pitch + offset, iw, ih, &ccol );
384  gl_renderRect( bx+pitch+3-iw, by+(nshow-cur_planet_sel)*pitch-ih + offset, iw, ih, &ccol );
385  gl_renderRect( bx+1, by+(nshow-cur_planet_sel-1)*pitch + offset, ih, iw, &ccol );
386  gl_renderRect( bx+1, by+(nshow-cur_planet_sel)*pitch-iw + offset, ih, iw, &ccol );
387  gl_renderRect( bx+pitch+3-ih, by+(nshow-cur_planet_sel-1)*pitch + offset, ih, iw, &ccol );
388  gl_renderRect( bx+pitch+3-ih, by+(nshow-cur_planet_sel)*pitch-iw + offset, ih, iw, &ccol );
389  cnt=0;
390  buf[0]='\0';
391  if (cur_planet_sel == 0) {
392  int infopos = 0;
393  int stars = MAX( array_size( bgImages )-1, 0 );
394  cnt+=scnprintf( &buf[cnt], sizeof(buf)-cnt, _("System: %s\n"), _(sys->name) );
395  /* display sun information */
396  cnt+=scnprintf( &buf[cnt], sizeof(buf)-cnt, n_("%d-star system\n", "%d-star system\n", stars), stars );
397 
398  /* Nebula. */
399  if (sys->nebu_density > 0. ) {
400  /* Volatility */
401  if (sys->nebu_volatility > 700.)
402  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Nebula: Volatile, ") );
403  else if (sys->nebu_volatility > 300.)
404  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Nebula: Dangerous, ") );
405  else if (sys->nebu_volatility > 0.)
406  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Nebula: Unstable, ") );
407  else
408  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Nebula: Stable, ") );
409

411  if (sys->nebu_density > 700.)
412  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Dense\n") );
413  else if (sys->nebu_density < 300.)
414  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Light\n") );
415  else
416  cnt += scnprintf( &buf[cnt], sizeof(buf)-cnt, _("Medium\n") );
