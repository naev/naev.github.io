68 static int osd_sortCompare( const void *arg1, const void *arg2 )
69 {
70  const OSD_t *osd1, *osd2;
71  int ret, i, m;
72 
73  osd1 = (OSD_t*)arg1;
74  osd2 = (OSD_t*)arg2;
75 
76  /* Compare priority. */
77  if (osd1->priority > osd2->priority)
78  return +1;
79  else if (osd1->priority < osd2->priority)
80  return -1;
81 
82  /* Compare name. */
83  ret = strcmp( osd1->title, osd2->title );
84  if (ret != 0)
85  return ret;
86 
87  /* Compare items. */
88  m = MIN(array_size(osd1->items), array_size(osd2->items));
89  for (i=0; i<m; i++) {
90  ret = strcmp( osd1->msg[i], osd2->msg[i] );
91  if (ret != 0)
92  return ret;
93  }
94 
95  /* Compare on length. */
96  if (array_size(osd1->items) > array_size(osd2->items))
97  return +1;
98  if (array_size(osd1->items) < array_size(osd2->items))
99  return -1;
100 
101  /* Compare ID. */
102  if (osd1->id > osd2->id)
103  return +1;
104  else if (osd1->id < osd2->id)
105  return -1;
106  return 0;
107 }

161 void osd_wordwrap( OSD_t* osd )
162 {
163  int i, n, l, s, w, t;
164  char *chunk;
165  for (i=0; i<array_size(osd->items); i++) {
166  for (l=0; l<array_size(osd->items[i]); l++)
167  free(osd->items[i][l]);
168  array_resize( &osd->items[i], 0 );
169 
170  l = strlen(osd->msg[i]); /* Message length. */
171  n = 0; /* Text position. */
172  t = 0; /* Tabbed? */
173  w = osd_w-osd_hyphenLen;
174  while (n < l) {
175  /* Test if tabbed. */
176  if (n==0) {
177  if (osd->msg[i][n] == '\t') {
178  t = 1;
179  w = osd_w - osd_tabLen;
180  }
181  else {
182  t = 0;
183  w = osd_w - osd_hyphenLen;
184  }
185  }
186 
187  /* Get text size. */
188  s = gl_printWidthForText( &gl_smallFont, &osd->msg[i][n], w, NULL );
189 
190  if (n==0 && t==1)
191  w -= osd_hyphenLen;
192 
193  /* Copy text over. */
194  if (n==0) {
195  if (t==1) {
196  chunk = malloc(s+4);
197  snprintf( chunk, s+4, " %s", &osd->msg[i][n+1] );
198  }
199  else {
200  chunk = malloc(s+3);
201  snprintf( chunk, s+3, "- %s", &osd->msg[i][n] );
202  }
203  }
204  else if (t==1) {
205  chunk = malloc(s+4);
206  snprintf( chunk, s+4, " %s", &osd->msg[i][n] );
207  }
208  else {
209  chunk = malloc(s+1);
210  snprintf( chunk, s+1, "%s", &osd->msg[i][n] );
211  }
212  array_push_back( &osd->items[i], chunk );
213 
214  /* Go to next line. */
215  n += s + 1;
216  }
217  }
218 }

398 void osd_render (void)
399 {
400  OSD_t *ll;
401  double p;
402  int i, j, k, l, m;

405  int *ignore;
406  int nignore;
407  int is_duplicate, duplicates;
408  char title[1024];
409 
410  /* Nothing to render. */
411  if (osd_list == NULL)
412  return;
413 
414  nignore = array_size(osd_list);
415  ignore = calloc( nignore, sizeof( int ) );
416 
417  /* Background. */
418  gl_renderRect( osd_x-5., osd_y-(osd_rh+5.), osd_w+10., osd_rh+10, &cBlackHilight );
419 
420  /* Render each thingy. */
421  p = osd_y-gl_smallFont.h;
422  l = 0;
423  for (k=0; k<array_size(osd_list); k++) {
424  if (ignore[k])
425  continue;
426 
427  ll = &osd_list[k];
428  x = osd_x;
429  w = osd_w;
430 
431  /* Check how many duplicates we have, mark duplicates for ignoring */
432  duplicates = 0;
433  for (m=k+1; m<array_size(osd_list); m++) {
434  if ((strcmp(osd_list[m].title, ll->title) == 0) &&
435  (array_size(osd_list[m].items) == array_size(ll->items)) &&
436  (osd_list[m].active == ll->active)) {
437  is_duplicate = 1;
438  for (i=osd_list[m].active; i<array_size(osd_list[m].items); i++) {
439  if (array_size(osd_list[m].items[i]) == array_size(ll->items[i])) {
440  for (j=0; j<array_size(osd_list[m].items[i]); j++) {
441  if (strcmp(osd_list[m].items[i][j], ll->items[i][j]) != 0 ) {
442  is_duplicate = 0;
443  break;
444  }
445  }
446  } else {
447  is_duplicate = 0;
448  }
449  if (!is_duplicate)
450  break;
451  }
452  if (is_duplicate) {
453  duplicates++;
454  ignore[m] = 1;
455  }
456  }
457  }
458 
459  /* Print title. */
460  if (duplicates > 0)
461  snprintf( title, sizeof(title), "%s (%d)", ll->title, duplicates + 1 );
462  else
463  strncpy( title, ll->title, sizeof(title)-1 );
464  title[sizeof(title)-1] = '\0';
465  gl_printMaxRaw( &gl_smallFont, w, x, p, NULL, -1., title);
466  p -= gl_smallFont.h + 5.;
467  l++;
468  if (l >= osd_lines) {
469  free(ignore);
470  return;
471  }
472 
473  /* Print items. */
474  for (i=ll->active; i<array_size(ll->items); i++) {
475  x = osd_x;
476  w = osd_w;
477  c = (i == (int)ll->active) ? &cFontWhite : &cFontGrey;
478  for (j=0; j<array_size(ll->items[i]); j++) {
479  gl_printMaxRaw( &gl_smallFont, w, x, p,
480  c, -1., ll->items[i][j] );
481  if (j==0) {
482  w = osd_w - osd_hyphenLen;
483  x = osd_x + osd_hyphenLen;
484  }
485  p -= gl_smallFont.h + 5.;
486  l++;
487  if (l >= osd_lines) {
488  free(ignore);
489  return;
490  }
491  }
492  }
493  }
494 
495  free(ignore);
496 }
