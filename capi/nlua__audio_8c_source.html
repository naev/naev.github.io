273 static int audioL_new( lua_State *L )
274 {
275  LuaAudio_t la;
276  LuaFile_t *lf;
277  const char *name;
278  SDL_RWops *rw;
279  double master;
280 
281  name = NULL;
282  rw = NULL;
283 
284  if (lua_isstring(L,1))
285  name = lua_tostring(L,1);
286  else if (lua_isfile(L,1)) {
287  lf = lua_tofile(L,1);
288  name = lf->path;
289  }
290  else
291  NLUA_INVALID_PARAMETER(L);
292 
293  memset( &la, 0, sizeof(LuaAudio_t) );
294  if (!sound_disabled) {
295  rw = PHYSFSRWOPS_openRead( name );
296  if (rw==NULL)
297  NLUA_ERROR(L,"Unable to open '%s'", name );
298 
299  soundLock();
300  alGenSources( 1, &la.source );
301 
302  la.buf = malloc( sizeof(LuaBuffer_t) );
303  la.buf->refcount = 1;
304  alGenBuffers( 1, &la.buf->buffer );
305  sound_al_buffer( &la.buf->buffer, rw, name );
306 
307  /* Attach buffer. */
308  alSourcei( la.source, AL_BUFFER, la.buf->buffer );
309 
310  /* Defaults. */
311  master = sound_getVolumeLog();
312  alSourcef( la.source, AL_GAIN, master );
313  /* The behaviour of sources depends on whether or not they are mono or
314  * stereo. In the case they are stereo, no position stuff is actually
315  * done. However, if they are mono, they are played with absolute
316  * position and the sound heard depends on the listener. We can disable
317  * this by setting AL_SOURCE_RELATIVE which puts the listener always at
318  * the origin, and then setting the source at the same origin. It should
319  * be noted that depending on the sound model this can be bad if it is
320  * not bounded. */
321  alSourcei( la.source, AL_SOURCE_RELATIVE, AL_TRUE );
322  alSource3f( la.source, AL_POSITION, 0., 0., 0. );
323  al_checkErr();
324  soundUnlock();
325 
326  SDL_RWclose( rw );
327  }
328 
329  lua_pushaudio(L, la);
330  return 1;
331 }
