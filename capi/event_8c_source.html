366 
367  /* Create the event. */

369  created++;
370  }
371

373  if (created)

385 static int event_parseXML( EventData *temp, const xmlNodePtr parent )
386 {
387  xmlNodePtr node, cur;
388  char *buf;
389 
390  memset( temp, 0, sizeof(EventData) );
391 
392  /* get the name */
393  xmlr_attr_strd(parent, "name", temp->name);
394  if (temp->name == NULL)
395  WARN(_("Event in %s has invalid or no name"), EVENT_DATA_PATH);
396 
397  node = parent->xmlChildrenNode;
398 
399  do { /* load all the data */
400 
401  /* Only check nodes. */
402  xml_onlyNodes(node);
403 
404  /* Trigger. */
405  if (xml_isNode(node,"trigger")) {
406  buf = xml_get(node);
407  if (buf == NULL)
408  WARN(_("Event '%s': Null trigger type."), temp->name);
409  else if (strcmp(buf,"enter")==0)
410  temp->trigger = EVENT_TRIGGER_ENTER;
411  else if (strcmp(buf,"land")==0)
412  temp->trigger = EVENT_TRIGGER_LAND;
413  else if (strcmp(buf,"load")==0)
414  temp->trigger = EVENT_TRIGGER_LOAD;
415  else if (strcmp(buf,"none")==0)
416  temp->trigger = EVENT_TRIGGER_NONE;
417  else
418  WARN(_("Event '%s' has invalid 'trigger' parameter: %s"), temp->name, buf);
419 
420  continue;
421  }
422 
423  /* Flags. */
424  else if (xml_isNode(node,"flags")) { /* set the various flags */
425  cur = node->children;
426  do {
427  xml_onlyNodes(cur);
428  if (xml_isNode(cur,"unique")) {
429  temp->flags |= EVENT_FLAG_UNIQUE;
430  continue;
431  }
432  WARN(_("Event '%s' has unknown flag node '%s'."), temp->name, cur->name);
433  } while (xml_nextNode(cur));
434  continue;
435  }
436 
437  /* Notes for the python mission mapping script. */
438  else if (xml_isNode(node,"notes")) continue;
439 
440  /* Condition. */
441  xmlr_strd(node,"cond",temp->cond);
442 
443  /* Get chance. */
444  xmlr_float(node,"chance",temp->chance);
445 
446  /* Get proirity. */
447  xmlr_int(node,"priority",temp->priority);
448 
449  DEBUG(_("Unknown node '%s' in event '%s'"), node->name, temp->name);
450  } while (xml_nextNode(node));
451 
452  /* Process. */
453  temp->chance /= 100.;
454 
455 #define MELEMENT(o,s) \
456  if (o) WARN(_("Event '%s' missing/invalid '%s' element"), temp->name, s)
457  MELEMENT((temp->trigger!=EVENT_TRIGGER_NONE) && (temp->chance==0.),"chance");
458  MELEMENT(temp->trigger==EVENT_TRIGGER_NULL,"trigger");
459 #undef MELEMENT
460 
461  return 0;
462 }

510 static int event_parseFile( const char* file )
511 {
512  size_t bufsize;
513  xmlNodePtr node;
514  xmlDocPtr doc;
515  char *filebuf;
516  const char *pos, *start_pos;
517  EventData *temp;
518 
519 #ifdef DEBUGGING
520  /* To check if event is valid. */
521  int ret;
522 #endif /* DEBUGGING */
523 
524  /* Load string. */
525  filebuf = ndata_read( file, &bufsize );
526  if (filebuf == NULL) {
527  WARN(_("Unable to read data from '%s'"), file);
528  return -1;
529  }
530 
531  /* Skip if no XML. */
532  pos = strnstr( filebuf, "</event>", bufsize );
533  if (pos==NULL) {
534  pos = strnstr( filebuf, "function create", bufsize );
535  if ((pos != NULL) && !strncmp(pos,"--common",bufsize))
536  WARN(_("Event '%s' has create function but no XML header!"), file);
537  free(filebuf);
538  return 0;
539  }
540 
541  /* Separate XML header and Lua. */
542  start_pos = strnstr( filebuf, "<?xml ", bufsize );
543  pos = strnstr( filebuf, "--]]", bufsize );
544  if (pos == NULL || start_pos == NULL) {
545  WARN(_("Event file '%s' has missing XML header!"), file);
546  return -1;
547  }
548 
549  /* Parse the header. */
550  doc = xmlParseMemory( start_pos, pos-start_pos );
551  if (doc == NULL) {
552  WARN(_("Unable to parse document XML header for Event '%s'"), file);
553  return -1;
554  }
555 
556  /* Get the root node. */
557  node = doc->xmlChildrenNode;
558  if (!xml_isNode(node,XML_EVENT_TAG)) {
559  WARN(_("Malformed '%s' file: missing root element '%s'"), file, XML_EVENT_TAG);
560  return -1;
561  }
562 
563  temp = &array_grow(&event_data);
564  event_parseXML( temp, node );
565  temp->lua = strdup(filebuf);
566  temp->sourcefile = strdup(file);
567 
568 #ifdef DEBUGGING
569  /* Check to see if syntax is valid. */
570  ret = luaL_loadbuffer(naevL, temp->lua, strlen(temp->lua), temp->name );
571  if (ret == LUA_ERRSYNTAX) {
572  WARN(_("Event Lua '%s' syntax error: %s"),
573  file, lua_tostring(naevL,-1) );
574  } else {
575  lua_pop(naevL, 1);
576  }
577 #endif /* DEBUGGING */
578 
579  /* Clean up. */
580  xmlFreeDoc(doc);
581  free(filebuf);
582 
583  return 0;
584 }
