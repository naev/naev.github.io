308 static void board_stealAmmo( unsigned int wdw, const char *str )
309 {

318  if (pilot_countAmmo(p) <= 0) {
319  player_message(_("#rThe ship has no ammo."));
320  return;
321  }
322  /* Player is already at max ammo */
323  if (pilot_countAmmo(player.p) >= pilot_maxAmmo(player.p)) {
324  player_message(_("#rYou are already at max ammo."));
325  return;
326  }
327  if (board_fail(wdw))
328  return;
329  /* Steal the ammo */
330  for (i=0; i<array_size(p->outfits); i++) {
331  target_outfit_slot = p->outfits[i];
332  if (target_outfit_slot == NULL)
333  continue;
334  target_outfit = target_outfit_slot->outfit;
335  if (target_outfit == NULL)
336  continue;
337  /* outfit isn't a launcher */
338  if (!outfit_isLauncher(target_outfit)) {
339  continue;
340  }
341  nammo = target_outfit_slot->u.ammo.quantity;
342  ammo = target_outfit_slot->u.ammo.outfit;
343  /* launcher has no ammo */
344  if (ammo == NULL)
345  continue;
346  if (nammo <= 0) {
347  continue;
348  }
349  for (x=0; x<array_size(player.p->outfits); x++) {
350  int nadded = 0;
351  player_outfit_slot = player.p->outfits[x];
352  if (player_outfit_slot == NULL)
353  continue;
354  player_outfit = player_outfit_slot->outfit;
355  if (player_outfit == NULL)
356  continue;
357  if (!outfit_isLauncher(player_outfit)) {
358  continue;
359  }
360  if (strcmp(ammo->name, player_outfit_slot->u.ammo.outfit->name) != 0) {
361  continue;
362  }
363  /* outfit's ammo matches; try to add to player and remove from target */
364  nadded = pilot_addAmmo(player.p, player_outfit_slot, ammo, nammo);
365  nammo -= nadded;
366  pilot_rmAmmo(p, target_outfit_slot, nadded);
367  nreloaded += nadded;
368  if (nadded > 0)
369  player_message(_("#oYou looted: %d %s"), nadded, _(ammo->name));
370  if (nammo <= 0) {
371  break;
372  }
373  }
374  if (nammo <= 0) {
375  continue;
376  }
377  }
378  if (nreloaded <= 0)
379  player_message(_("#rThere is no ammo compatible with your launchers on board."));
380  pilot_updateMass(player.p);
381  pilot_weaponSafe(player.p);
382  pilot_updateMass(p);
383  pilot_weaponSafe(p);
384  board_update(wdw);
385 }
