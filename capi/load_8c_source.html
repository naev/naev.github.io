102 static int load_load( nsave_t *save, const char *path )
103 {
104  xmlDocPtr doc;
105  xmlNodePtr root, parent, node, cur;
106  int cycles, periods, seconds;
107 
108  memset( save, 0, sizeof(nsave_t) );
109 
110  /* Load the XML. */
111  doc = load_xml_parsePhysFS( path );
112  if (doc == NULL) {
113  WARN( _("Unable to parse save path '%s'."), path);
114  return -1;
115  }
116  root = doc->xmlChildrenNode; /* base node */
117  if (root == NULL) {
118  WARN( _("Unable to get child node of save '%s'."), path);
119  xmlFreeDoc(doc);
120  return -1;
121  }
122 
123  /* Save path. */
124  save->path = strdup(path);
125 
126  /* Iterate inside the naev_save. */
127  parent = root->xmlChildrenNode;
128  do {
129  xml_onlyNodes(parent);
130 
131  /* Info. */
132  if (xml_isNode(parent, "version")) {
133  node = parent->xmlChildrenNode;
134  do {
135  xmlr_strd(node, "naev", save->version);
136  xmlr_strd(node, "data", save->data);
137  } while (xml_nextNode(node));
138  continue;
139  }
140 
141  else if (xml_isNode(parent, "player")) {
142  /* Get name. */
143  xmlr_attr_strd(parent, "name", save->name);
144  /* Parse rest. */
145  node = parent->xmlChildrenNode;
146  do {
147  xml_onlyNodes(node);
148 
149  /* Player info. */
150  xmlr_strd(node, "location", save->planet);
151  xmlr_ulong(node, "credits", save->credits);
152 
153  /* Time. */
154  if (xml_isNode(node, "time")) {
155  cur = node->xmlChildrenNode;
156  cycles = periods = seconds = 0;
157  do {
158  xmlr_int(cur, "SCU", cycles);
159  xmlr_int(cur, "STP", periods);
160  xmlr_int(cur, "STU", seconds);
161  } while (xml_nextNode(cur));
162  save->date = ntime_create( cycles, periods, seconds );
163  continue;
164  }
165 
166  /* Ship info. */
167  if (xml_isNode(node, "ship")) {
168  xmlr_attr_strd(node, "name", save->shipname);
169  xmlr_attr_strd(node, "model", save->shipmodel);
170  continue;
171  }
172  } while (xml_nextNode(node));
173  continue;
174  }
175  } while (xml_nextNode(parent));
176 
177  /* Clean up. */
178  xmlFreeDoc(doc);
179 
180  return 0;
181 }

187 int load_refresh (void)
188 {
189  char buf[PATH_MAX];
190  filedata_t *files, tmp;
191  size_t len;
192  int i, ok;
193  nsave_t *ns;
194 
195  if (load_saves != NULL)
196  load_free();
197 
198  /* load the saves */
199  files = array_create( filedata_t );
200  PHYSFS_enumerate( "saves", load_enumerateCallback, &files );
201  qsort(files, array_size(files), sizeof(filedata_t), load_sortCompare);
202 
203  if (array_size(files) == 0) {
204  array_free( files );
205  return 0;
206  }
207 
208  /* Make sure backups are after saves. */
209  for (i=0; i<array_size(files)-1; i++) {
210  len = strlen( files[i].name );
211 
212  /* Only interested in swapping backup with file after it if it's not backup. */
213  if ((len < 11) || strcmp( &files[i].name[len-10],".ns.backup" ))
214  continue;
215 
216  /* Don't match. */
217  if (strncmp( files[i].name, files[i+1].name, len-10 ))
218  continue;
219 
220  /* Swap around. */
221  tmp = files[i];
222  files[i] = files[i+1];
223  files[i+1] = tmp;
224  }
225 
226  /* Allocate and parse. */
227  ok = 0;
228  ns = NULL;
229  load_saves = array_create_size( nsave_t, array_size(files) );
230  for (i=0; i<array_size(files); i++) {
231  if (!ok)
232  ns = &array_grow( &load_saves );
233  snprintf( buf, sizeof(buf), "saves/%s", files[i].name );
234  ok = load_load( ns, buf );
235  }
236 
237  /* If the save was invalid, array is 1 member too large. */
238  if (ok)
239  array_resize( &load_saves, array_size(load_saves)-1 );
240 
241  /* Clean up memory. */
242  for (i=0; i<array_size(files); i++)
243  free( files[i].name );
244  array_free( files );
245 
246  return 0;
247 }

253 static int load_enumerateCallback( void* data, const char* origdir, const char* fname )
254 {
255  char *path;
256  const char *fmt;
257  size_t dir_len, name_len;
258  filedata_t *tmp;
259  PHYSFS_Stat stat;
260 
261  dir_len = strlen( origdir );
262  name_len = strlen( fname );
263 
264  /* no save or backup save extension? */
265  if ((name_len < 4 || strcmp( &fname[name_len-3], ".ns" )) && (name_len < 11 || strcmp( &fname[name_len-10], ".ns.backup" )))
266  return PHYSFS_ENUM_OK;
267 
268  fmt = dir_len && origdir[dir_len-1]=='/' ? "%s%s" : "%s/%s";
269  asprintf( &path, fmt, origdir, fname );
270  if (!PHYSFS_stat( path, &stat ))
271  WARN( _("PhysicsFS: Cannot stat %s: %s"), path,
272  PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
273  else if (stat.filetype == PHYSFS_FILETYPE_REGULAR) {
274  tmp = &array_grow( (filedata_t**)data );
275  tmp->name = strdup( fname );
276  tmp->stat = stat;
277  }
278 
279  free( path );
280  return PHYSFS_ENUM_OK;
281 }

634 static int load_gameInternal( const char* file, const char* version )
635 {

644  return -1;
645  }
646 
647  /* Load the XML. */
648  doc = load_xml_parsePhysFS( file );
649  if (doc == NULL)
650  goto err;
651  node = doc->xmlChildrenNode; /* base node */
652  if (node == NULL)
653  goto err_doc;
654 
655  /* Clean up possible stuff that should be cleaned. */
656  player_cleanup();

666  pnt = player_load(node);

668

701  land( pnt, 1 );
702

706  gui_load( player.p->ship->gui );
707  }

719
