242 static int planetL_getBackend( lua_State *L, int landable )
243 {
244  int i;
245  int *factions;
246  char **planets;
247  const char *rndplanet;
248  LuaSystem luasys;
249  Planet *pnt;
250  StarSystem *sys;
251  char *sysname;
252 
253  rndplanet = NULL;
254  planets = NULL;
255 
256  /* If boolean return random. */
257  if (lua_isboolean(L,1)) {
258  pnt = planet_get( space_getRndPlanet(landable, 0, NULL) );
259  lua_pushplanet(L,planet_index( pnt ));
260  luasys = system_index( system_get( planet_getSystem(pnt->name) ) );
261  lua_pushsystem(L,luasys);
262  return 2;
263  }
264 
265  /* Get a planet by faction */
266  else if (lua_isfaction(L,1)) {
267  factions = array_create( int );
268  array_push_back( &factions, lua_tofaction(L,1) );
269  planets = space_getFactionPlanet( factions, landable );
270  array_free( factions );
271  }
272 
273  /* Get a planet by name */
274  else if (lua_isstring(L,1)) {
275  rndplanet = lua_tostring(L,1);
276 
277  if (landable) {
278  pnt = planet_get( rndplanet );
279  if (pnt == NULL) {
280  NLUA_ERROR(L, _("Planet '%s' not found in stack"), rndplanet);
281  return 0;
282  }
283 
284  /* Check if can land. */
285  planet_updateLand( pnt );
286  if (!pnt->can_land)
287  return 0;
288  }
289  }
290 
291  /* Get a planet from faction list */
292  else if (lua_istable(L,1)) {
293  /* Get table length and preallocate. */
294  factions = array_create_size( int, lua_objlen(L,1) );
295  /* Load up the table. */
296  lua_pushnil(L);
297  while (lua_next(L, -2) != 0) {
298  if (lua_isfaction(L, -1))
299  array_push_back( &factions, lua_tofaction(L, -1) );
300  lua_pop(L,1);
301  }
302 
303  /* get the planets */
304  planets = space_getFactionPlanet( factions, landable );
305  array_free(factions);
306  }
307  else
308  NLUA_INVALID_PARAMETER(L); /* Bad Parameter */
309 
310  /* Pick random planet */
311  if (rndplanet == NULL) {
312  arrayShuffle( (void**)planets );
313 
314  for (i=0; i<array_size(planets); i++) {
315  if (landable) {
316  /* Check landing. */
317  pnt = planet_get( planets[i] );
318  if (pnt == NULL)
319  continue;
320 
321  planet_updateLand( pnt );
322  if (!pnt->can_land)
323  continue;
324  }
325 
326  rndplanet = planets[i];
327  break;
328  }
329  }
330  array_free(planets);
331 
332  /* No suitable planet found */
333  if (rndplanet == NULL && array_size( planets ) == 0)
334  return 0;
335 
336  /* Push the planet */
337  pnt = planet_get(rndplanet); /* The real planet */
338  if (pnt == NULL) {
339  NLUA_ERROR(L, _("Planet '%s' not found in stack"), rndplanet);
340  return 0;
341  }
342  sysname = planet_getSystem(rndplanet);
343  if (sysname == NULL) {
344  NLUA_ERROR(L, _("Planet '%s' is not placed in a system"), rndplanet);
345  return 0;
346  }
347  sys = system_get( sysname );
348  if (sys == NULL) {
349  NLUA_ERROR(L, _("Planet '%s' can't find system '%s'"), rndplanet, sysname);
350  return 0;
351  }
352  lua_pushplanet(L,planet_index( pnt ));
353  luasys = system_index( sys );
354  lua_pushsystem(L,luasys);
355  return 2;
356 }
