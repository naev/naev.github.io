246 static int planetL_getBackend( lua_State *L, int landable )
247 {
248  int i;
249  int *factions;
250  char **planets;
251  const char *rndplanet;
252  LuaSystem luasys;
253  Planet *pnt;
254  StarSystem *sys;
255  char *sysname;
256 
257  rndplanet = NULL;
258  planets = NULL;
259 
260  /* If boolean return random. */
261  if (lua_isboolean(L,1)) {
262  pnt = planet_get( space_getRndPlanet(landable, 0, NULL) );
263  lua_pushplanet(L,planet_index( pnt ));
264  luasys = system_index( system_get( planet_getSystem(pnt->name) ) );
265  lua_pushsystem(L,luasys);
266  return 2;
267  }
268 
269  /* Get a planet by faction */
270  else if (lua_isfaction(L,1)) {
271  factions = array_create( int );
272  array_push_back( &factions, lua_tofaction(L,1) );
273  planets = space_getFactionPlanet( factions, landable );
274  array_free( factions );
275  }
276 
277  /* Get a planet by name */
278  else if (lua_isstring(L,1)) {
279  rndplanet = lua_tostring(L,1);
280 
281  if (landable) {
282  pnt = planet_get( rndplanet );
283  if (pnt == NULL) {
284  NLUA_ERROR(L, _("Planet '%s' not found in stack"), rndplanet);
285  return 0;
286  }
287 
288  /* Check if can land. */
289  planet_updateLand( pnt );
290  if (!pnt->can_land)
291  return 0;
292  }
293  }
294 
295  /* Get a planet from faction list */
296  else if (lua_istable(L,1)) {
297  /* Get table length and preallocate. */
298  factions = array_create_size( int, lua_objlen(L,1) );
299  /* Load up the table. */
300  lua_pushnil(L);
301  while (lua_next(L, -2) != 0) {
302  if (lua_isfaction(L, -1))
303  array_push_back( &factions, lua_tofaction(L, -1) );
304  lua_pop(L,1);
305  }
306 
307  /* get the planets */
308  planets = space_getFactionPlanet( factions, landable );
309  array_free(factions);
310  }
311  else
312  NLUA_INVALID_PARAMETER(L); /* Bad Parameter */
313 
314  /* Pick random planet */
315  if (rndplanet == NULL) {
316  arrayShuffle( (void**)planets );
317 
318  for (i=0; i<array_size(planets); i++) {
319  if (landable) {
320  /* Check landing. */
321  pnt = planet_get( planets[i] );
322  if (pnt == NULL)
323  continue;
324 
325  planet_updateLand( pnt );
326  if (!pnt->can_land)
327  continue;
328  }
329 
330  rndplanet = planets[i];
331  break;
332  }
333  }
334  array_free(planets);
335 
336  /* No suitable planet found */
337  if (rndplanet == NULL && array_size( planets ) == 0)
338  return 0;
339 
340  /* Push the planet */
341  pnt = planet_get(rndplanet); /* The real planet */
342  if (pnt == NULL) {
343  NLUA_ERROR(L, _("Planet '%s' not found in stack"), rndplanet);
344  return 0;
345  }
346  sysname = planet_getSystem(rndplanet);
347  if (sysname == NULL) {
348  NLUA_ERROR(L, _("Planet '%s' is not placed in a system"), rndplanet);
349  return 0;
350  }
351  sys = system_get( sysname );
352  if (sys == NULL) {
353  NLUA_ERROR(L, _("Planet '%s' can't find system '%s'"), rndplanet, sysname);
354  return 0;
355  }
356  lua_pushplanet(L,planet_index( pnt ));
357  luasys = system_index( sys );
358  lua_pushsystem(L,luasys);
359  return 2;
360 }
