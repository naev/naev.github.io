245 static int planetL_getBackend( lua_State *L, int system, int landable )
246 {
247  int *factions;
248  char **planets;
249  const char *rndplanet;
250  LuaSystem luasys;
251  Planet *pnt;
252  StarSystem *sys;
253  char *sysname;
254 
255  rndplanet = NULL;
256  planets = NULL;
257 
258  /* If boolean return random. */
259  if (lua_isboolean(L,1)) {
260  pnt = planet_get( space_getRndPlanet(landable, 0, NULL) );
261  lua_pushplanet(L,planet_index( pnt ));
262  luasys = system_index( system_get( planet_getSystem(pnt->name) ) );
263  lua_pushsystem(L,luasys);
264  return 2;
265  }
266 
267  /* Get a planet by faction */
268  else if (lua_isfaction(L,1)) {
269  factions = array_create( int );
270  array_push_back( &factions, lua_tofaction(L,1) );
271  planets = space_getFactionPlanet( factions, landable );
272  array_free( factions );
273  }
274 
275  /* Get a planet by name */
276  else if (lua_isstring(L,1)) {
277  rndplanet = lua_tostring(L,1);
278 
279  if (landable) {
280  pnt = planet_get( rndplanet );
281  if (pnt == NULL) {
282  NLUA_ERROR(L, _("Planet '%s' not found in stack"), rndplanet);
283  return 0;
284  }
285 
286  /* Check if can land. */
287  planet_updateLand( pnt );
288  if (!pnt->can_land)
289  return 0;
290  }
291  }
292 
293  /* Get a planet from faction list */
294  else if (lua_istable(L,1)) {
295  /* Get table length and preallocate. */
296  factions = array_create_size( int, lua_objlen(L,1) );
297  /* Load up the table. */
298  lua_pushnil(L);
299  while (lua_next(L, -2) != 0) {
300  if (lua_isfaction(L, -1))
301  array_push_back( &factions, lua_tofaction(L, -1) );
302  lua_pop(L,1);
303  }
304 
305  /* get the planets */
306  planets = space_getFactionPlanet( factions, landable );
307  array_free(factions);
308  }
309 
310  /* Just get a planet. */
311  else if (lua_isplanet(L,1)) {
312  pnt = luaL_validplanet( L, 1 );
313  if (landable) {
314  /* Check if can land. */
315  planet_updateLand( pnt );
316  if (!pnt->can_land)
317  return 0;
318  }
319  rndplanet = pnt->name;
320  }
321 
322  else
323  NLUA_INVALID_PARAMETER(L); /* Bad Parameter */
324 
325  /* Pick random planet */
326  if (rndplanet == NULL) {
327  arrayShuffle( (void**)planets );
328 
329  for (int i=0; i<array_size(planets); i++) {
330  if (landable) {
331  /* Check landing. */
332  pnt = planet_get( planets[i] );
333  if (pnt == NULL)
334  continue;
335 
336  planet_updateLand( pnt );
337  if (!pnt->can_land)
338  continue;
339  }
340 
341  rndplanet = planets[i];
342  break;
343  }
344  }
345  array_free(planets);
346 
347  /* No suitable planet found */
348  if (rndplanet == NULL && array_size( planets ) == 0)
349  return 0;
350 
351  /* Push the planet */
352  pnt = planet_get(rndplanet); /* The real planet */
353  if (pnt == NULL) {
354  NLUA_ERROR(L, _("Planet '%s' not found in stack"), rndplanet);
355  return 0;
356  }
357  sysname = planet_getSystem(rndplanet);
358  if (sysname == NULL) {
359  NLUA_ERROR(L, _("Planet '%s' is not placed in a system"), rndplanet);
360  return 0;
361  }
362  sys = system_get( sysname );
363  if (sys == NULL) {
364  NLUA_ERROR(L, _("Planet '%s' can't find system '%s'"), rndplanet, sysname);
365  return 0;
366  }
367  lua_pushplanet(L,planet_index( pnt ));
368  luasys = system_index( sys );
369  if (system) {
370  lua_pushsystem(L,luasys);
371  return 2;
372  }
373  return 1;
374 }
