92 void background_initStars( int n )
93 {
94  unsigned int i;
95  GLfloat w, h, hw, hh;
96  double size;
97  GLfloat *star_vertex;
98 
99  /* Calculate size. */
100  size = SCREEN_W*SCREEN_H+STAR_BUF*STAR_BUF;
101  size /= pow2(conf.zoom_far);
102 
103  /* Calculate star buffer. */
104  w = (SCREEN_W + 2.*STAR_BUF);
105  w += conf.zoom_stars * (w / conf.zoom_far - 1.);
106  h = (SCREEN_H + 2.*STAR_BUF);
107  h += conf.zoom_stars * (h / conf.zoom_far - 1.);
108  hw = w / 2.;
109  hh = h / 2.;
110 
111  /* Calculate stars. */
112  size *= n;
113  nstars = (unsigned int)(size/(800.*600.));
114 
115  /* Create data. */
116  star_vertex = malloc( nstars * sizeof(GLfloat) * 6 );
117 
118  for (i=0; i < nstars; i++) {
119  /* Set the position. */
120  star_vertex[6*i+0] = RNGF()*w - hw;
121  star_vertex[6*i+1] = RNGF()*h - hh;
122  star_vertex[6*i+3] = star_vertex[6*i+0];
123  star_vertex[6*i+4] = star_vertex[6*i+1];
124  /* Set the colour. */
125  star_vertex[6*i+2] = RNGF()*0.6 + 0.2;
126  star_vertex[6*i+5] = star_vertex[6*i+2];
127  }
128 
129  /* Recreate VBO. */
130  gl_vboDestroy( star_vertexVBO );
131  star_vertexVBO = gl_vboCreateStatic(
132  nstars * sizeof(GLfloat) * 6, star_vertex );
133 
134  free(star_vertex);
135 }

156 void background_renderStars( const double dt )
157 {
158  (void) dt;
159  GLfloat h, w;
160  GLfloat x, y, m;
161  double z;
162  gl_Matrix4 projection;
163 
164  glUseProgram(shaders.stars.program);
165 
166  glLineWidth(1 / gl_screen.scale);
167 
168  /* Do some scaling for now. */
169  z = cam_getZoom();
170  z = 1. * (1. - conf.zoom_stars) + z * conf.zoom_stars;
171  projection = gl_Matrix4_Translate( gl_view_matrix, SCREEN_W/2., SCREEN_H/2., 0 );
172  projection = gl_Matrix4_Scale( projection, z, z, 1 );
173 
174  /* Decide on shade mode. */
175  x = 0;
176  y = 0;
177  if ((player.p != NULL) && !player_isFlag(PLAYER_DESTROYED) &&
178  !player_isFlag(PLAYER_CREATING)) {
179 
180  if (pilot_isFlag(player.p,PILOT_HYPERSPACE)) { /* hyperspace fancy effects */
181  /* lines get longer the closer we are to finishing the jump */
182  m = MAX( 0, HYPERSPACE_STARS_BLUR-player.p->ptimer );
183  m /= HYPERSPACE_STARS_BLUR;
184  m *= HYPERSPACE_STARS_LENGTH;
185  x = m*cos(VANGLE(player.p->solid->vel));
186  y = m*sin(VANGLE(player.p->solid->vel));
187  }
188  else if (dt_mod * VMOD(player.p->solid->vel) > 500. ) {
189  /* Very short lines tend to flicker horribly. A stock Llama at 2x
190  * speed just so happens to make very short lines. A 5px minimum
191  * is long enough to (mostly) alleviate the flickering.
192  */
193  m = MAX( 5, dt_mod*VMOD(player.p->solid->vel)/25. - 20 );
194  x = m*cos(VANGLE(player.p->solid->vel));
195  y = m*sin(VANGLE(player.p->solid->vel));
196  }
197  }
198 
199  /* Calculate some dimensions. */
200  w = (SCREEN_W + 2.*STAR_BUF);
201  w += conf.zoom_stars * (w / conf.zoom_far - 1.);
202  h = (SCREEN_H + 2.*STAR_BUF);
203  h += conf.zoom_stars * (h / conf.zoom_far - 1.);
204 
205  /* Render. */
206  glEnableVertexAttribArray( shaders.stars.vertex );
207  glEnableVertexAttribArray( shaders.stars.brightness );
208  gl_vboActivateAttribOffset( star_vertexVBO, shaders.stars.vertex, 0,
209  2, GL_FLOAT, 3 * sizeof(GLfloat) );
210  gl_vboActivateAttribOffset( star_vertexVBO, shaders.stars.brightness, 2 * sizeof(GLfloat),
211  1, GL_FLOAT, 3 * sizeof(GLfloat) );
212 
213  gl_Matrix4_Uniform(shaders.stars.projection, projection);
214  glUniform2f(shaders.stars.star_xy, star_x, star_y);
215  glUniform2f(shaders.stars.wh, w, h);
216  glUniform2f(shaders.stars.xy, x, y);
217  glUniform1f(shaders.stars.scale, 1 / gl_screen.scale);
218  glDrawArrays( GL_LINES, 0, nstars );
219 
220  /* Disable vertex array. */
221  glDisableVertexAttribArray( shaders.stars.vertex );
222  glDisableVertexAttribArray( shaders.stars.brightness );
223 
224  glLineWidth(1 / gl_screen.scale);
225 
226  glUseProgram(0);
227 
228  /* Check for errors. */
229  gl_checkErr();
230 }

333 static void background_renderImages( background_image_t *bkg_arr )
334 {
335  int i;
336  background_image_t *bkg;
337  double px,py, x,y, xs,ys, z;
338  glColour col;
339 
340  /* Skip rendering altogether if disabled. */
341  if (conf.bg_brightness <= 0.)
342  return;
343 
344  /* Render images in order. */
345  for (i=0; i<array_size(bkg_arr); i++) {
346  bkg = &bkg_arr[i];
347 
348  cam_getPos( &px, &py );
349  x = px + (bkg->x - px) * bkg->move - bkg->scale*bkg->image->sw/2.;
350  y = py + (bkg->y - py) * bkg->move - bkg->scale*bkg->image->sh/2.;
351  gl_gameToScreenCoords( &xs, &ys, x, y );
352  z = cam_getZoom();
353  z *= bkg->scale;
354  /* TODO add rotation too! */
355  col.r = bkg->col.r * conf.bg_brightness;
356  col.g = bkg->col.g * conf.bg_brightness;
357  col.b = bkg->col.b * conf.bg_brightness;
358  col.a = bkg->col.a;
359  gl_blitScale( bkg->image, xs, ys,
360  z*bkg->image->sw, z*bkg->image->sh, &col );
361  }
362 }
