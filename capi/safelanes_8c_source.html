167 {

171 
172  out = array_create( SafeLane );
173 
174  for (i=sys_to_first_edge[system->id]; i<sys_to_first_edge[1+system->id]; i++) {
175  for (j=0; j<2; j++)

247 static double safelanes_changeFromOptimizer (void)
248 {
249  cholmod_sparse *stiff_s;
250  cholmod_factor *stiff_f;
251  cholmod_dense *new_utilde, *_QtQutilde, *Lambda_tilde;
252  double rel_change = HUGE_VAL;
253  double zero[] = {0, 0}, neg_1[] = {-1, 0};
254 
255  stiff_s = cholmod_triplet_to_sparse( stiff, 0, &C );
256  stiff_f = cholmod_analyze( stiff_s, &C );
257  cholmod_factorize( stiff_s, stiff_f, &C );
258  new_utilde = cholmod_solve( CHOLMOD_A, stiff_f, ftilde, &C );
259  if (utilde != NULL) {
260  /* Do utilde -= new_utilde, and compute ||utilde||/||utilde_new|| */
261  for (size_t i = 0; i < utilde->nzmax; i++)
262  ((double*)utilde->x)[i] -= ((double*)new_utilde->x)[i];
263  rel_change = frobenius_norm( utilde ) / frobenius_norm( new_utilde );
264  cholmod_free_dense( &utilde, &C );
265  }
266  utilde = new_utilde;
267 
268  if (rel_change > CONVERGENCE_THRESHOLD) {
269  _QtQutilde = cholmod_zeros( utilde->nrow, utilde->ncol, CHOLMOD_REAL, &C );
270  cholmod_sdmult( QtQ, 0, neg_1, zero, utilde, _QtQutilde, &C );
271  Lambda_tilde = cholmod_solve( CHOLMOD_A, stiff_f, _QtQutilde, &C );
272  cholmod_free_dense( &_QtQutilde, &C );
273  safelanes_activateByGradient( Lambda_tilde );
274  cholmod_free_dense( &Lambda_tilde, &C );
275  }
276  cholmod_free_factor( &stiff_f, &C );
277  cholmod_free_sparse( &stiff_s, &C );
278  return rel_change;
279 }

346 static void safelanes_initStacks_edge (void)
347 {
348  const StarSystem *systems_stack;
349  int system, i, j, k;
350 
351  systems_stack = system_getAll();
352 
353  edge_stack = array_create( Edge );
354  sys_to_first_edge = array_create( int );
355  array_push_back( &sys_to_first_edge, 0 );
356  lane_fmask = array_create( FactionMask );
357  tmp_edge_conduct = array_create( double );
358  for (system=0; system<array_size(systems_stack); system++) {
359  for (i = sys_to_first_vertex[system]; i < sys_to_first_vertex[1+system]; i++) {
360  const Vector2d *pi = vertex_pos( i );
361  for (j = sys_to_first_vertex[system]; j < i; j++) {
362  const Vector2d *pj = vertex_pos( j );
363  double lij = vect_dist( pi, pj );
364  int has_approx_midpoint = 0;
365  for (k = sys_to_first_vertex[system]; k < sys_to_first_vertex[1+system]; k++)
366  if (k!=i && k!=j && safelanes_triangleTooFlat( pi, pj, vertex_pos( k ), lij )) {
367  has_approx_midpoint = 1;
368  break;
369  }
370  if (has_approx_midpoint)
371  continue; /* The edge from i to j is disallowed in favor of a path through k. */
372  array_push_back_edge( &edge_stack, j, i );
373  array_push_back( &lane_fmask, MASK_COMPROMISE( vertex_faction( i ), vertex_faction( j ) ) );
374  array_push_back( &tmp_edge_conduct, 1/lij );
375  }
376  }
377  array_push_back( &sys_to_first_edge, array_size(edge_stack) );
378  }
379  array_shrink( &edge_stack );
380  array_shrink( &sys_to_first_edge );
381 
382  lane_faction = array_create_size( int, array_size(edge_stack) );
383  memset( lane_faction, 0, array_size(lane_faction)*sizeof(lane_faction[0]) );
384 }

475 static void safelanes_initStiff (void)
476 {
477  int nnz, v, i;
478  int *pi, *pj;
479  double *pv;
480  double max_conductivity;
481 
482  cholmod_free_triplet( &stiff, &C );
483  v = array_size(vertex_stack);
484  nnz = 3*(array_size(edge_stack)+array_size(tmp_jump_edges)) + array_size(tmp_anchor_vertices);
485  stiff = cholmod_allocate_triplet( v, v, nnz, STORAGE_MODE_UPPER_TRIANGULAR_PART, CHOLMOD_REAL, &C );
486  /* Populate triplets: internal edges (ii ij jj), implicit jump connections (ditto), anchor conditions. */
487  pi = stiff->i; pj = stiff->j; pv = stiff->x;
488  for (i=0; i<array_size(edge_stack); i++) {
489  *pi++ = edge_stack[i][0]; *pj++ = edge_stack[i][0]; *pv++ = +tmp_edge_conduct[i];
490  *pi++ = edge_stack[i][0]; *pj++ = edge_stack[i][1]; *pv++ = -tmp_edge_conduct[i];
491  *pi++ = edge_stack[i][1]; *pj++ = edge_stack[i][1]; *pv++ = +tmp_edge_conduct[i];
492  }
493  for (i=0; i<array_size(tmp_jump_edges); i++) {
494  *pi++ = tmp_jump_edges[i][0]; *pj++ = tmp_jump_edges[i][0]; *pv++ = +JUMP_CONDUCTIVITY;
495  *pi++ = tmp_jump_edges[i][0]; *pj++ = tmp_jump_edges[i][1]; *pv++ = -JUMP_CONDUCTIVITY;
496  *pi++ = tmp_jump_edges[i][1]; *pj++ = tmp_jump_edges[i][1]; *pv++ = +JUMP_CONDUCTIVITY;
497  }
498  /* Add a Robin boundary condition, using the max conductivity (after activation) for spectral reasons. */
499  max_conductivity = JUMP_CONDUCTIVITY/(ALPHA+1);
500  for (i=0; i<array_size(edge_stack); i++)
501  max_conductivity = MAX( max_conductivity, tmp_edge_conduct[i] );
502  max_conductivity = MAX( JUMP_CONDUCTIVITY, (1+ALPHA)*max_conductivity ); /* Activation scales entries by 1+ALPHA later. */
503  for (i=0; i<array_size(tmp_anchor_vertices); i++) {
504  *pi++ = tmp_anchor_vertices[i]; *pj++ = tmp_anchor_vertices[i]; *pv++ = max_conductivity;
505  }
506  stiff->nnz = nnz;
507 #if DEBUGGING
508  assert( cholmod_check_triplet( stiff, &C) );
509 #endif
510 }

687 static int safelanes_triangleTooFlat( const Vector2d* m, const Vector2d* n, const Vector2d* p, double lmn )
688 {
689  const double MAX_COSINE = cos(MIN_ANGLE);
690  double lnp = vect_dist( n, p );
691  double lmp = vect_dist( m, p );
692  double dpn = ((n->x-m->x)*(n->x-p->x) + (n->y-m->y)*(n->y-p->y)) / ( lmn * lnp );
693  double dpm = ((m->x-n->x)*(m->x-p->x) + (m->y-n->y)*(m->y-p->y)) / ( lmn * lmp );
694  return (dpn > MAX_COSINE && lnp < lmn) || (dpm > MAX_COSINE && lmp < lmn);
695 }

767 {
