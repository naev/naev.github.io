273 {

277  double coef = luaL_checknumber(L,4);
278  const char *skind = luaL_optstring(L,5,"real");
279  int haslb, hasub, type, kind;
280  double lb, ub;
281 
282  /* glpk stuff */
283  glp_set_col_name( lp->prob, idx, name );
284  glp_set_obj_coef( lp->prob, idx, coef );
285 
286  /* Determine bounds. */
287  haslb = !lua_isnoneornil(L,6);
288  hasub = !lua_isnoneornil(L,7);
289  lb = luaL_optnumber(L,6,0.0);
290  ub = luaL_optnumber(L,7,0.0);
291  if (haslb && hasub) {
292  if (fabs(lb-ub) < 1e-5)
293  type = GLP_FX;
294  else
295  type = GLP_DB;
296  }
297  else if (haslb)
298  type = GLP_LO;
299  else if (hasub)
300  type = GLP_UP;
301  else
302  type = GLP_FR;
303  glp_set_col_bnds( lp->prob, idx, type, lb, ub );
304 
305  /* Get kind. */
306  if (strcmp(skind,"real")==0)
307  kind = GLP_CV;
308  else if (strcmp(skind,"integer")==0)
309  kind = GLP_IV;
310  else if (strcmp(skind,"binary")==0)
311  kind = GLP_BV;
312  else
313  NLUA_ERROR(L,_("Unknown column kind '%s'!"), skind);
314  glp_set_col_kind( lp->prob, idx, kind );
315 
316  return 0;
317 }
