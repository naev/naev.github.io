508 static int pilotL_add( lua_State *L )
509 {
510  Ship *ship;
511  const char *shipname, *pilotname, *ai;
512  int i;
513  unsigned int p;
514  double a, r;
515  Vector2d vv, vp, vn;
516  LuaFaction lf;
517  StarSystem *ss;
518  Planet *planet;
519  JumpPoint *jump;
520  PilotFlags flags;
521  int ignore_rules;
522  Pilot *pplt;
523 
524  NLUA_CHECKRW(L);
525 
526  /* Default values. */
527  pilot_clearFlagsRaw( flags );
528  vectnull(&vn); /* Need to determine angle. */
529  jump = NULL;
530  planet = NULL;
531  a = 0.;
532 
533  /* Parse first argument - Ship Name */
534  shipname = luaL_checkstring(L,1);
535 
536  /* pull the fleet */
537  ship = NULL;
538  ship = ship_get( shipname );
539  if (ship == NULL) {
540  NLUA_ERROR(L,_("Ship '%s' not found!"), shipname);
541  return 0;
542  }
543  /* Get pilotname argument if provided. */
544  pilotname = luaL_optstring( L, 4, shipname );
545  /* Get faction from string or number. */
546  lf = luaL_validfaction(L,2);
547 
548  /* Handle position/origin argument. */
549  if (lua_isvector(L,3)) {
550  vp = *lua_tovector(L,3);
551  a = RNGF() * 2.*M_PI;
552  vectnull( &vv );
553  }
554  else if (lua_issystem(L,3)) {
555  ss = system_getIndex( lua_tosystem(L,3) );
556  for (i=0; i<array_size(cur_system->jumps); i++) {
557  if ((cur_system->jumps[i].target == ss)
558  && !jp_isFlag( cur_system->jumps[i].returnJump, JP_EXITONLY )) {
559  jump = cur_system->jumps[i].returnJump;
560  break;
561  }
562  }
563  if (jump == NULL) {
564  if (array_size(cur_system->jumps) > 0) {
565  WARN(_("Ship '%s' jumping in from non-adjacent system '%s' to '%s'."),
566  pilotname, ss->name, cur_system->name );
567  jump = cur_system->jumps[RNG_BASE(0, array_size(cur_system->jumps)-1)].returnJump;
568  }
569  else
570  WARN(_("Ship '%s' attempting to jump in from '%s', but '%s' has no jump points."),
571  pilotname, ss->name, cur_system->name );
572  }
573  }
574  else if (lua_isplanet(L,3)) {
575  planet = luaL_validplanet(L,3);
576  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
577  a = RNGF() * 2. * M_PI;
578  r = RNGF() * planet->radius;
579  vect_cset( &vp,
580  planet->pos.x + r * cos(a),
581  planet->pos.y + r * sin(a) );
582  a = RNGF() * 2.*M_PI;
583  vectnull( &vv );
584  }
585  /* Random. */
586  else {
587  /* Check if we should ignore the strict rules. */
588  ignore_rules = 0;
589  if (lua_isboolean(L,3) && lua_toboolean(L,3))
590  ignore_rules = 1;
591 
592  /* Choose the spawn point and act in consequence.*/
593  pilot_choosePoint( &vp, &planet, &jump, lf, ignore_rules, 0 );
594 
595  if (planet != NULL) {
596  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
597  a = RNGF() * 2. * M_PI;
598  r = RNGF() * planet->radius;
599  vect_cset( &vp,
600  planet->pos.x + r * cos(a),
601  planet->pos.y + r * sin(a) );
602  a = RNGF() * 2.*M_PI;
603  vectnull( &vv );
604  }
605  else {
606  a = RNGF() * 2.*M_PI;
607  vectnull( &vv );
608  }
609  }
610 
611  /* Parse final argument - table of optional parameters */
612  ai = NULL;
613  if (lua_gettop( L ) >= 5 && !lua_isnil( L, 5 )) {
614  if (!lua_istable( L, 5 )) {
615  NLUA_ERROR( L, _("'parameters' should be a table of options or omitted!") );
616  return 0;
617  }
618  lua_getfield( L, 5, "ai" );
619  ai = luaL_optstring( L, -1, NULL );
620  lua_pop( L, 1 );
621 
622  lua_getfield( L, 5, "naked" );
623  if (lua_toboolean(L, -1))
624  pilot_setFlagRaw( flags, PILOT_NO_OUTFITS );
625  lua_pop( L, 1 );
626 
627  lua_getfield( L, 5, "stealth" );
628  if (lua_toboolean(L, -1))
629  pilot_setFlagRaw( flags, PILOT_STEALTH );
630  lua_pop( L, 1 );
631  }
632 
633  /* Set up velocities and such. */
634  if (jump != NULL) {
635  space_calcJumpInPos( cur_system, jump->from, &vp, &vv, &a, NULL );
636  pilot_setFlagRaw( flags, PILOT_HYP_END );
637  }
638 
639  /* Make sure angle is valid. */
640  a = fmod( a, 2.*M_PI );
641  if (a < 0.)
642  a += 2.*M_PI;
643 
644  /* Create the pilot. */
645  p = pilot_create( ship, pilotname, lf, ai, a, &vp, &vv, flags, 0, 0 );
646  lua_pushpilot(L,p);
647  pplt = pilot_get( p );
648 
649  /* Set the memory stuff. */
650  if (jump != NULL) {
651  LuaJump lj;
652  lj.srcid = jump->from->id;
653  lj.destid = cur_system->id;
654 
655  nlua_getenv( pplt->ai->env, AI_MEM );
656  lua_pushjump(L, lj);
657  lua_setfield(L,-2,"create_jump");
658  lua_pop(L,1);
659  }
660  else if (planet != NULL) {
661  nlua_getenv( pplt->ai->env, AI_MEM );
662  lua_pushplanet(L,planet->id);
663  lua_setfield(L,-2,"create_planet");
664  lua_pop(L,1);
665  }
666 
667  /* TODO don't have space_calcJumpInPos called twice when stealth creating. */
668  if ((jump != NULL) && pilot_isFlagRaw( flags, PILOT_STEALTH )) {
669  space_calcJumpInPos( cur_system, jump->from, &pplt->solid->pos, &pplt->solid->vel, &pplt->solid->dir, pplt );
670  }
671  return 1;
672 }
