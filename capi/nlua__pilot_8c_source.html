504 static int pilotL_add( lua_State *L )
505 {
506  Ship *ship;
507  const char *shipname, *pilotname, *ai;
508  int i;
509  unsigned int p;
510  double a, r;
511  Vector2d vv, vp, vn;
512  LuaFaction lf;
513  StarSystem *ss;
514  Planet *planet;
515  JumpPoint *jump;
516  PilotFlags flags;
517  int ignore_rules;
518  Pilot *pplt;
519 
520  NLUA_CHECKRW(L);
521 
522  /* Default values. */
523  pilot_clearFlagsRaw( flags );
524  vectnull(&vn); /* Need to determine angle. */
525  jump = NULL;
526  a = 0.;
527 
528  /* Parse first argument - Ship Name */
529  shipname = luaL_checkstring(L,1);
530 
531  /* pull the fleet */
532  ship = NULL;
533  ship = ship_get( shipname );
534  if (ship == NULL) {
535  NLUA_ERROR(L,_("Ship '%s' not found!"), shipname);
536  return 0;
537  }
538  /* Get pilotname argument if provided. */
539  pilotname = luaL_optstring( L, 4, shipname );
540  /* Get faction from string or number. */
541  lf = luaL_validfaction(L,2);
542 
543  /* Handle position/origin argument. */
544  if (lua_isvector(L,3)) {
545  vp = *lua_tovector(L,3);
546  a = RNGF() * 2.*M_PI;
547  vectnull( &vv );
548  }
549  else if (lua_issystem(L,3)) {
550  ss = system_getIndex( lua_tosystem(L,3) );
551  for (i=0; i<array_size(cur_system->jumps); i++) {
552  if ((cur_system->jumps[i].target == ss)
553  && !jp_isFlag( cur_system->jumps[i].returnJump, JP_EXITONLY )) {
554  jump = cur_system->jumps[i].returnJump;
555  break;
556  }
557  }
558  if (jump == NULL) {
559  if (array_size(cur_system->jumps) > 0) {
560  WARN(_("Ship '%s' jumping in from non-adjacent system '%s' to '%s'."),
561  pilotname, ss->name, cur_system->name );
562  jump = cur_system->jumps[RNG_BASE(0, array_size(cur_system->jumps)-1)].returnJump;
563  }
564  else
565  WARN(_("Ship '%s' attempting to jump in from '%s', but '%s' has no jump points."),
566  pilotname, ss->name, cur_system->name );
567  }
568  }
569  else if (lua_isplanet(L,3)) {
570  planet = luaL_validplanet(L,3);
571  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
572  a = RNGF() * 2. * M_PI;
573  r = RNGF() * planet->radius;
574  vect_cset( &vp,
575  planet->pos.x + r * cos(a),
576  planet->pos.y + r * sin(a) );
577  a = RNGF() * 2.*M_PI;
578  vectnull( &vv );
579  }
580  /* Random. */
581  else {
582  /* Check if we should ignore the strict rules. */
583  ignore_rules = 0;
584  if (lua_isboolean(L,3) && lua_toboolean(L,3))
585  ignore_rules = 1;
586 
587  /* Choose the spawn point and act in consequence.*/
588  pilot_choosePoint( &vp, &planet, &jump, lf, ignore_rules, 0 );
589 
590  if (planet != NULL) {
591  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
592  a = RNGF() * 2. * M_PI;
593  r = RNGF() * planet->radius;
594  vect_cset( &vp,
595  planet->pos.x + r * cos(a),
596  planet->pos.y + r * sin(a) );
597  a = RNGF() * 2.*M_PI;
598  vectnull( &vv );
599  }
600  else {
601  a = RNGF() * 2.*M_PI;
602  vectnull( &vv );
603  }
604  }
605 
606  /* Parse final argument - table of optional parameters */
607  ai = NULL;
608  if (lua_gettop( L ) >= 5 && !lua_isnil( L, 5 )) {
609  if (!lua_istable( L, 5 )) {
610  NLUA_ERROR( L, _("'parameters' should be a table of options or omitted!") );
611  return 0;
612  }
613  lua_getfield( L, 5, "ai" );
614  ai = luaL_optstring( L, -1, NULL );
615  lua_pop( L, 1 );
616 
617  lua_getfield( L, 5, "naked" );
618  if (lua_toboolean(L, -1))
619  pilot_setFlagRaw( flags, PILOT_NO_OUTFITS );
620  lua_pop( L, 1 );
621 
622  lua_getfield( L, 5, "stealth" );
623  if (lua_toboolean(L, -1))
624  pilot_setFlagRaw( flags, PILOT_STEALTH );
625  lua_pop( L, 1 );
626  }
627 
628  /* Set up velocities and such. */
629  if (jump != NULL) {
630  space_calcJumpInPos( cur_system, jump->from, &vp, &vv, &a, NULL );
631  pilot_setFlagRaw( flags, PILOT_HYP_END );
632  }
633 
634  /* Make sure angle is valid. */
635  a = fmod( a, 2.*M_PI );
636  if (a < 0.)
637  a += 2.*M_PI;
638 
639  /* Create the pilot. */
640  p = pilot_create( ship, pilotname, lf, ai, a, &vp, &vv, flags, 0, 0 );
641  lua_pushpilot(L,p);
642 
643  /* TODO don't have space_calcJumpInPos called twice when stealth creating. */
644  if ((jump != NULL) && pilot_isFlagRaw( flags, PILOT_STEALTH )) {
645  pplt = pilot_get( p );
646  space_calcJumpInPos( cur_system, jump->from, &pplt->solid->pos, &pplt->solid->vel, &pplt->solid->dir, pplt );
647  }
648  return 1;
649 }
