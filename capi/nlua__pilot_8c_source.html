503 static int pilotL_add( lua_State *L )
504 {
505  const Ship *ship;
506  const char *pilotname, *ai;
507  unsigned int p;
508  double a, r;
509  Vector2d vv, vp, vn;
510  LuaFaction lf;
511  StarSystem *ss;
512  Planet *planet;
513  JumpPoint *jump;
514  PilotFlags flags;
515  int ignore_rules;
516  Pilot *pplt;
517 
518  NLUA_CHECKRW(L);
519 
520  /* Default values. */
521  pilot_clearFlagsRaw( flags );
522  vectnull(&vn); /* Need to determine angle. */
523  ss = NULL;
524  jump = NULL;
525  planet= NULL;
526  a = 0.;
527 
528  /* Parse first argument - Ship Name */
529  ship = luaL_validship(L,1);
530  /* Get faction from string or number. */
531  lf = luaL_validfaction(L,2);
532  /* Get pilotname argument if provided. */
533  pilotname = luaL_optstring( L, 4, ship->name );
534 
535  /* Handle position/origin argument. */
536  if (lua_isvector(L,3)) {
537  vp = *lua_tovector(L,3);
538  a = RNGF() * 2.*M_PI;
539  vectnull( &vv );
540  }
541  else if (lua_issystem(L,3))
542  ss = system_getIndex( lua_tosystem(L,3) );
543  else if (lua_isjump(L,3))
544  ss = system_getIndex( lua_tojump(L,3)->destid );
545  else if (lua_isplanet(L,3)) {
546  planet = luaL_validplanet(L,3);
547  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
548  a = RNGF() * 2. * M_PI;
549  r = RNGF() * planet->radius;
550  vect_cset( &vp,
551  planet->pos.x + r * cos(a),
552  planet->pos.y + r * sin(a) );
553  a = RNGF() * 2.*M_PI;
554  vectnull( &vv );
555  }
556  /* Random. */
557  else if (lua_isnoneornil(L,3)) {
558  /* Check if we should ignore the strict rules. */
559  ignore_rules = 0;
560  if (lua_isboolean(L,3) && lua_toboolean(L,3))
561  ignore_rules = 1;
562 
563  /* Choose the spawn point and act in consequence.*/
564  pilot_choosePoint( &vp, &planet, &jump, lf, ignore_rules, 0 );
565 
566  if (planet != NULL) {
567  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
568  a = RNGF() * 2. * M_PI;
569  r = RNGF() * planet->radius;
570  vect_cset( &vp,
571  planet->pos.x + r * cos(a),
572  planet->pos.y + r * sin(a) );
573  a = RNGF() * 2.*M_PI;
574  vectnull( &vv );
575  }
576  else {
577  a = RNGF() * 2.*M_PI;
578  vectnull( &vv );
579  }
580  }
581  else
582  NLUA_INVALID_PARAMETER(L);
583 
584  /* Handle system. */
585  if (ss != NULL) {
586  for (int i=0; i<array_size(cur_system->jumps); i++) {
587  if ((cur_system->jumps[i].target == ss)
588  && !jp_isFlag( cur_system->jumps[i].returnJump, JP_EXITONLY )) {
589  jump = cur_system->jumps[i].returnJump;
590  break;
591  }
592  }
593  if (jump == NULL) {
594  if (array_size(cur_system->jumps) > 0) {
595  WARN(_("Ship '%s' jumping in from non-adjacent system '%s' to '%s'."),
596  pilotname, ss->name, cur_system->name );
597  jump = cur_system->jumps[RNG_BASE(0, array_size(cur_system->jumps)-1)].returnJump;
598  }
599  else
600  WARN(_("Ship '%s' attempting to jump in from '%s', but '%s' has no jump points."),
601  pilotname, ss->name, cur_system->name );
602  }
603  }
604 
605  /* Parse final argument - table of optional parameters */
606  ai = NULL;
607  if (lua_gettop( L ) >= 5 && !lua_isnil( L, 5 )) {
608  if (!lua_istable( L, 5 )) {
609  NLUA_ERROR( L, _("'parameters' should be a table of options or omitted!") );
610  return 0;
611  }
612  lua_getfield( L, 5, "ai" );
613  ai = luaL_optstring( L, -1, NULL );
614  lua_pop( L, 1 );
615 
616  lua_getfield( L, 5, "naked" );
617  if (lua_toboolean(L, -1))
618  pilot_setFlagRaw( flags, PILOT_NO_OUTFITS );
619  lua_pop( L, 1 );
620 
621  lua_getfield( L, 5, "stealth" );
622  if (lua_toboolean(L, -1))
623  pilot_setFlagRaw( flags, PILOT_STEALTH );
624  lua_pop( L, 1 );
625  }
626 
627  /* Set up velocities and such. */
628  if (jump != NULL) {
629  space_calcJumpInPos( cur_system, jump->from, &vp, &vv, &a, NULL );
630  pilot_setFlagRaw( flags, PILOT_HYP_END );
631  }
632 
633  /* Make sure angle is valid. */
634  a = fmod( a, 2.*M_PI );
635  if (a < 0.)
636  a += 2.*M_PI;
637 
638  /* Create the pilot. */
639  p = pilot_create( ship, pilotname, lf, ai, a, &vp, &vv, flags, 0, 0 );
640  lua_pushpilot(L,p);
641  pplt = pilot_get( p );
642 
643  /* Set the memory stuff. */
644  if (jump != NULL) {
645  LuaJump lj;
646  lj.srcid = jump->from->id;
647  lj.destid = cur_system->id;
648 
649  nlua_getenv( pplt->ai->env, AI_MEM );
650  lua_pushjump(L, lj);
651  lua_setfield(L,-2,"create_jump");
652  lua_pop(L,1);
653  }
654  else if (planet != NULL) {
655  nlua_getenv( pplt->ai->env, AI_MEM );
656  lua_pushplanet(L,planet->id);
657  lua_setfield(L,-2,"create_planet");
658  lua_pop(L,1);
659  }
660 
661  /* TODO don't have space_calcJumpInPos called twice when stealth creating. */
662  if ((jump != NULL) && pilot_isFlagRaw( flags, PILOT_STEALTH )) {
663  space_calcJumpInPos( cur_system, jump->from, &pplt->solid->pos, &pplt->solid->vel, &pplt->solid->dir, pplt );
664  }
665  return 1;
666 }
