1263 static int playerL_teleport( lua_State *L )

1268

1270  PLAYER_CHECK();

1273  if (landed)
1274  NLUA_ERROR(L,_("Can not teleport the player while landed!"));
1275  if (comm_isOpen())
1276  NLUA_ERROR(L,_("Can not teleport the player while the comm is open!"));
1277  if (player_isBoarded())
1278  NLUA_ERROR(L,_("Can not teleport the player while they are boarded!"));
1279 
1280  pnt = NULL;
1281 
1282  /* Get a system. */
1283  if (lua_issystem(L,1)) {
1284  sys = luaL_validsystem(L,1);
1285  name = system_getIndex(sys->id)->name;
1286  }
1287  /* Get a planet. */
1288  else if (lua_isplanet(L,1)) {
1289  pnt = luaL_validplanet(L,1);
1290  name = planet_getSystem( pnt->name );
1291  if (name == NULL) {
1292  NLUA_ERROR( L, _("Planet '%s' does not belong to a system."), pnt->name );
1293  return 0;
1294  }
1295  }
1296  /* Get destination from string. */
1297  else if (lua_isstring(L,1)) {
1298  name = lua_tostring(L,1);
1299  sysname = system_existsCase( name );
1300  if (sysname == NULL) {
1301  /* No system found, assume destination string is the name of a planet. */
1302  pntname = name;
1303  name = planet_getSystem( pntname );
1304  pnt = planet_get( pntname );
1305  if (pnt == NULL) {
1306  NLUA_ERROR( L, _("'%s' is not a valid teleportation target."), name );
1307  return 0;
1308  }
1309 
1310  if (name == NULL) {
1311  NLUA_ERROR( L, _("Planet '%s' does not belong to a system."), pntname );
1312  return 0;
1313  }
1314  }
1315  else
1316  sys = system_get( sysname );
1317  }
1318  else
1319  NLUA_INVALID_PARAMETER(L);
1320 
1321  /* Check if system exists. */
1322  if (system_get( name ) == NULL) {
1323  NLUA_ERROR( L, _("System '%s' does not exist."), name );
1324  return 0;
1325  }
1326 
1327  /* Jump out hook is run first. */
1328  hooks_run( "jumpout" );
1329 
1330  /* Just in case remove hyperspace flags. */
1331  pilot_rmFlag( player.p, PILOT_HYPERSPACE );
1332  pilot_rmFlag( player.p, PILOT_HYP_BEGIN );
1333  pilot_rmFlag( player.p, PILOT_HYP_BRAKE );
1334  pilot_rmFlag( player.p, PILOT_HYP_PREP );
1335  /* Don't follow anything. */
1336  player_accelOver();
1337  player_autonavEnd();
1338 
1339  /* Free graphics. */
1340  space_gfxUnload( cur_system );
1341

1343  * Both of these functions invoke gui_setNav(), which updates jump and
1344  * planet targets simultaneously. Thus, invalid reads may arise and the

1346  * the old system.
1347  */

1349

1351  space_init( name );
1352

1354  map_clear();
1355

1357  ovr_initAlpha();

1361  hooks_run( "enter" );
